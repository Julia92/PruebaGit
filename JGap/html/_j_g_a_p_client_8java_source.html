168  protected void evolve(GridClient gc)
169  throws Exception {
170  // Do the complete evolution cycle until end.
171  // ------------------------------------------
172  IClientFeedback feedback = m_gridConfig.getClientFeedback();
173  feedback.beginWork();
174  IClientEvolveStrategy evolver = m_gridConfig.getClientEvolveStrategy();
175  IRequestSplitStrategy splitter = m_gridConfig.getRequestSplitStrategy();
176  int evolutionIndex = 0;
177  do {
178 // m_clientEvolveStrategy.beforeGenerateWorkResults();
179  JGAPRequest[] workRequests = evolver.generateWorkRequests(
180  m_workReq, splitter, null);
181  feedback.setProgressMaximum(0);
182  feedback.setProgressMaximum(workRequests.length - 1);
183  sendWorkRequests(workRequests);
184  if (this.isInterrupted()) {
185  break;
186  }
187  evolver.afterWorkRequestsSent();
188  receiveWorkResults(workRequests);
189  evolver.evolve();
190  // Fire listener that one evolution cycle is complete.
191  // ---------------------------------------------------
192  feedback.completeFrame(evolutionIndex);
193  evolutionIndex++;
194  // Check if evolution is finished.
195  // -------------------------------
196  if (evolver.isEvolutionFinished(evolutionIndex)) {
197  evolver.onFinished();
198  break;
199  }
200  } while (true);
201  m_gridConfig.getClientFeedback().endWork();
