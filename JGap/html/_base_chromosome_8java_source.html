483  // ------------
484  int index = 0;
485  StringTokenizer st0 = new StringTokenizer

491  a.add(fitnessS);
492  index += fitnessS.length();

494  throw new UnsupportedRepresentationException("Number of genes expected!");
495  }

497  a.add(numGenes);

501

503  throw new UnsupportedRepresentationException("Gene data missing!");
504  }

507  // ---------------------------------
508  a_string = a_string.substring(index);

511  // ----------
512  StringTokenizer st = new StringTokenizer

514  while (st.hasMoreTokens()) {
