123  public void selectChromosomes(final int a_howManyToSelect,
124  Population a_to_pop) {
125  List tournament = new Vector();
126  RandomGenerator rn = getConfiguration().getRandomGenerator();
127  int size = m_chromosomes.size();
128  if (size == 0) {
129  return;
130  }
131  int k;
132  for (int i = 0; i < a_howManyToSelect; i++) {
133  // Choose [tournament size] individuals from the population at random.
134  // -------------------------------------------------------------------
135  tournament.clear();
136  for (int j = 0; j < m_config.m_tournament_size; j++) {
137  k = rn.nextInt(size);
138  tournament.add(m_chromosomes.get(k));
139  }
140  Collections.sort(tournament, m_fitnessValueComparator);
141  double prob = rn.nextDouble();
142  double probAccumulated = m_config.m_probability;
143  int index = 0;
144  // Play the tournament.
145  // --------------------
146  if (m_config.m_tournament_size > 1) {
147  do {
148  if (prob <= probAccumulated) {
149  break;
150  }
151  else {
152  probAccumulated += probAccumulated * (1 - m_config.m_probability);
153  index++;
154  }
155  } while (index < m_config.m_tournament_size - 1);
156  }
157  a_to_pop.addChromosome( (IChromosome) tournament.get(index));
158  }
159  }
