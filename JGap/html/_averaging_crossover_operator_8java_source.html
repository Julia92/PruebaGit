164  public void operate(final Population a_population,
165  final List a_candidateChromosomes) {
166  // Determine the number of crossovers that should be performed
167  int size = Math.min(getConfiguration().getPopulationSize(),
168  a_population.size());
169  int numCrossovers = 0;
170  if (m_crossoverRateCalc == null) {
171  numCrossovers = size / m_crossoverRate;
172  }
173  else {
174  numCrossovers = size / m_crossoverRateCalc.calculateCurrentRate();
175  }
176  RandomGenerator generator = getConfiguration().getRandomGenerator();
177  if (m_crossoverGenerator == null) {
178  m_crossoverGenerator = generator;
179  }
180  // For each crossover, grab two random chromosomes, pick a random
181  // locus (gene location), and then swap that gene and all genes
182  // to the "right" (those with greater loci) of that gene between
183  // the two chromosomes.
184  // --------------------------------------------------------------
185  int index1, index2;
186  for (int i = 0; i < numCrossovers; i++) {
187  index1 = generator.nextInt(size);
188  index2 = generator.nextInt(size);
189  IChromosome origChrom1 = a_population.getChromosome(index1);
190  IChromosome origChrom2 = a_population.getChromosome(index2);
191  IChromosome firstMate = (IChromosome)origChrom1.clone();
192  IChromosome secondMate = (IChromosome)origChrom2.clone();
193  // In case monitoring is active, support it.
194  // -----------------------------------------
195  if (m_monitorActive) {
196  firstMate.setUniqueIDTemplate(origChrom1.getUniqueID(), 1);
197  firstMate.setUniqueIDTemplate(origChrom2.getUniqueID(), 2);
198  secondMate.setUniqueIDTemplate(origChrom1.getUniqueID(), 1);
199  secondMate.setUniqueIDTemplate(origChrom2.getUniqueID(), 2);
200  }
201  Gene[] firstGenes = firstMate.getGenes();
202  Gene[] secondGenes = secondMate.getGenes();
203  int locus = getLocus(m_crossoverGenerator, i, firstGenes.length);
204  // Swap the genes.
205  // ---------------
206  Gene gene1;
207  Gene gene2;
208  Object firstAllele;
209  for (int j = locus; j < firstGenes.length; j++) {
210  // Make a distinction to ICompositeGene for the first gene.
211  // --------------------------------------------------------
212  if (firstGenes[j] instanceof ICompositeGene) {
213  // Randomly determine gene to be considered
214  index1 = generator.nextInt(firstGenes[j].size());
215  gene1 = ( (ICompositeGene) firstGenes[j]).geneAt(index1);
216  }
217  else {
218  gene1 = firstGenes[j];
219  }
220  // Make a distinction to ICompositeGene for the second gene.
221  // ---------------------------------------------------------
222  if (secondGenes[j] instanceof CompositeGene) {
223  // Randomly determine gene to be considered
224  index2 = generator.nextInt(secondGenes[j].size());
225  gene2 = ( (ICompositeGene) secondGenes[j]).geneAt(index2);
226  }
227  else {
228  gene2 = secondGenes[j];
229  }
230  firstAllele = gene1.getAllele();
231  gene1.setAllele(gene2.getAllele());
232  gene2.setAllele(firstAllele);
233  }
234  // Add the modified chromosomes to the candidate pool so that
235  // they'll be considered for natural selection during the next
236  // phase of evolution.
237  // -----------------------------------------------------------
238  a_candidateChromosomes.add(firstMate);
239  a_candidateChromosomes.add(secondMate);
240  }
241  }
