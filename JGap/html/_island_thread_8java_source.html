36  public IslandThread(int nextNumber)
37  throws Exception {
38  m_nextNumber = nextNumber;
39  String threadKey = Thread.currentThread().getId() + "/" + m_nextNumber;
40  System.out.println("Starting thread: " + nextNumber);
41  Configuration conf = new DefaultConfiguration(threadKey, threadKey);
42  FitnessFunction myFunc =
43  new MinimizingMakeChangeFitnessFunction(93);
44  conf.setFitnessFunction(myFunc);
45  // Now we need to tell the Configuration object how we want our
46  // Chromosomes to be setup. We do that by actually creating a
47  // sample Chromosome and then setting it on the Configuration
48  // object. As mentioned earlier, we want our Chromosomes to each
49  // have four genes, one for each of the coin types. We want the
50  // values (alleles) of those genes to be integers, which represent
51  // how many coins of that type we have. We therefore use the
52  // IntegerGene class to represent each of the genes. That class
53  // also lets us specify a lower and upper bound, which we set
54  // to sensible values for each coin type.
55  // --------------------------------------------------------------
56  Gene[] sampleGenes = new Gene[4];
57  sampleGenes[0] = new IntegerGene(conf, 0, 3 * 10); // Quarters
58  sampleGenes[1] = new IntegerGene(conf, 0, 2 * 10); // Dimes
59  sampleGenes[2] = new IntegerGene(conf, 0, 1 * 10); // Nickels
60  sampleGenes[3] = new IntegerGene(conf, 0, 4 * 10); // Pennies
61  IChromosome sampleChromosome = new Chromosome(conf, sampleGenes);
62  conf.setSampleChromosome(sampleChromosome);
63  // Finally, we need to tell the Configuration object how many
64  // Chromosomes we want in our population. The more Chromosomes,
65  // the larger number of potential solutions (which is good for
66  // finding the answer), but the longer it will take to evolve
67  // the population (which could be seen as bad).
68  // ------------------------------------------------------------
69  conf.setPopulationSize(80);
70  gen = Genotype.randomInitialGenotype(conf);
71  }
