210  protected int solve(Configuration a_conf, int a_targetChangeAmount,
211  SupergeneChangeFitnessFunction a_fitnessFunction,
212  Gene[] a_sampleGenes)
213  throws InvalidConfigurationException {
214  IChromosome sampleChromosome = new Chromosome(a_conf, a_sampleGenes);
215  a_conf.setSampleChromosome(sampleChromosome);
216  // Finally, we need to tell the Configuration object how many
217  // Chromosomes we want in our population. The more Chromosomes,
218  // the larger number of potential solutions (which is good for
219  // finding the answer), but the longer it will take to evolve
220  // the population (which could be seen as bad). We'll just set
221  // the population size to 500 here.
222  // ------------------------------------------------------------
223  a_conf.setPopulationSize(POPULATION_SIZE);
224  // Create random initial population of Chromosomes.
225  // ------------------------------------------------
226  Genotype population = Genotype.randomInitialGenotype(a_conf);
227  int s;
228  Evolution:
229  // Evolve the population, break if the the change solution is found.
230  // -----------------------------------------------------------------
231  for (int i = 0; i < MAX_ALLOWED_EVOLUTIONS; i++) {
232  population.evolve();
233  s = Math.abs(a_fitnessFunction.amountOfChange(population.
234  getFittestChromosome())
235  - a_targetChangeAmount);
236  if (s == 0) {
237  break Evolution;
238  }
239  }
240  // Display the best solution we found.
241  // -----------------------------------
242  IChromosome bestSolutionSoFar = report(a_fitnessFunction, population);
243  return Math.abs(a_fitnessFunction.amountOfChange(bestSolutionSoFar)
244  - a_targetChangeAmount);
