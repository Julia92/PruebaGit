516  public static Double[][] transposeMatrix(Double[][] m) {
517  int r = m.length;
518  int c = m[0].length;
519  Double[][] t = new Double[c][r];
520  for (int i = 0; i < r; ++i) {
521  for (int j = 0; j < c; ++j) {
522  t[j][i] = m[i][j];
523  }
524  }
525  return t;
526  } // end transposeMatrix
