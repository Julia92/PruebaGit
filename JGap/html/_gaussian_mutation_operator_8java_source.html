88  final List a_candidateChromosomes) {
89  int size = Math.min(getConfiguration().getPopulationSize(),
90  a_population.size());
91  if (m_rg == null) {
92  RandomGenerator rn = getConfiguration().getRandomGenerator();
93  m_rg = rn;
94  }
95  for (int i = 0; i < size; i++) {
96  Gene[] genes = a_population.getChromosome(i).getGenes();
97  IChromosome originalChrom = a_population.getChromosome(i);
98  IChromosome copyOfChromosome = null;
99  // For each Chromosome in the population...
100  // ----------------------------------------
101  for (int j = 0; j < genes.length; j++) {
102  double nextGaussian = m_rg.nextDouble();
103  double diff = nextGaussian * m_deviation;
104  // ...take a copy of it...
105  // -----------------------
106  if (copyOfChromosome == null) {
107  copyOfChromosome = (IChromosome) originalChrom.clone();
108  // ...add it to the candidate pool...
109  // ----------------------------------
110  a_candidateChromosomes.add(copyOfChromosome);
111  // ...then Gaussian mutate all its genes
112  genes = copyOfChromosome.getGenes();
113  // In case monitoring is active, support it.
114  // -----------------------------------------
115  if (m_monitorActive) {
116  copyOfChromosome.setUniqueIDTemplate(originalChrom.getUniqueID(), 1);
117  }
118  }
119  // Process all atomic elements in the gene. For a StringGene this
120  // would be the length of the string, for an IntegerGene, it is
121  // always one element.
122  // --------------------------------------------------------------
123  if (genes[j] instanceof CompositeGene) {
124  CompositeGene compositeGene = (CompositeGene) genes[j];
125  if (m_monitorActive) {
126  compositeGene.setUniqueIDTemplate(originalChrom.getGene(j).
127  getUniqueID(), 1);
128  }
129  for (int k = 0; k < compositeGene.size(); k++) {
130  mutateGene(compositeGene.geneAt(k), diff);
131  if (m_monitorActive) {
132  compositeGene.geneAt(k).setUniqueIDTemplate(
133  ( (ICompositeGene) originalChrom.getGene(j)).geneAt(k).
134  getUniqueID(), 1);
135  }
136  }
137  }
138  else {

140  if (m_monitorActive) {

142  1);
143  }
144  }
145  }
