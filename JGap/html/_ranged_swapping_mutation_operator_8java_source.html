134  public void operate(final Population a_population,
135  List a_candidateChromosomes) {
136  // this was a private variable, now it is local reference.
137  final IUniversalRateCalculator m_mutationRateCalc = getMutationRateCalc();
138  // If the mutation rate is set to zero and dynamic mutation rate is
139  // disabled, then we don't perform any mutation.
140  // ----------------------------------------------------------------
141  if (getMutationRate() == 0 && m_mutationRateCalc == null) {
142  return;
143  }
144  // Determine the mutation rate. If dynamic rate is enabled, then
145  // calculate it based upon the number of genes in the chromosome.
146  // Otherwise, go with the mutation rate set upon construction.
147  // --------------------------------------------------------------
148  int currentRate;
149  if (m_mutationRateCalc != null) {
150  currentRate = m_mutationRateCalc.calculateCurrentRate();
151  }
152  else {
153  currentRate = getMutationRate();
154  }
155  RandomGenerator generator = getConfiguration().getRandomGenerator();
156  // It would be inefficient to create copies of each Chromosome just
157  // to decide whether to mutate them. Instead, we only make a copy
158  // once we've positively decided to perform a mutation.
159  // ----------------------------------------------------------------
160  int size = a_population.size();
161  for (int i = 0; i < size; i++) {
162  IChromosome x = a_population.getChromosome(i);
163  // This returns null if not mutated:
164  IChromosome xm = operate(x, currentRate, generator);
165  if (xm != null) {
166  a_candidateChromosomes.add(xm);
167  }
168  }
169  }

183  protected IChromosome operate(final IChromosome a_chrom, final int a_rate,
184  final RandomGenerator a_generator) {
185  IChromosome chromosome = null;
186  // ----------------------------------------
187  for (int j = m_startOffset; j < a_chrom.size(); j++) {
188  // Ensure probability of 1/currentRate for applying mutation.
189  // ----------------------------------------------------------
190  if (a_generator.nextInt(a_rate) == 0) {
191  if (chromosome == null) {
192  chromosome = (IChromosome) a_chrom.clone();
193  // In case monitoring is active, support it.
194  // -----------------------------------------
195  if (m_monitorActive) {
196  chromosome.setUniqueIDTemplate(a_chrom.getUniqueID(), 1);
197  }
198  }
199  Gene[] genes = chromosome.getGenes();
200  if (m_range == 0) {
201  m_range = genes.length;
202  }
203  Gene[] mutated = operate(a_generator, j, genes);
204  // setGenes is not required for this operator, but it may
205  // be needed for the derived operators.
206  // ------------------------------------------------------
207  try {
208  chromosome.setGenes(mutated);
209  } catch (InvalidConfigurationException cex) {
210  throw new Error("Gene type not allowed by constraint checker", cex);
211  }
212  }
213  }
214  return chromosome;
215  }

231  protected Gene[] operate(final RandomGenerator a_generator,
232  final int a_target_gene, final Gene[] a_genes) {
233  // Other needs to be an integer that is within the range of the subject
234  // target gene.
235 
236  int rand = a_generator.nextInt(2 * m_range);
237  int other = (a_target_gene - m_range) + rand;
238  if (other < 0) {
239  other = 0;
240  }
241  if (other >= a_genes.length) {
242  other = a_genes.length - 1; // Index is -1 of length
243  }
244  Gene t = a_genes[a_target_gene];
245  a_genes[a_target_gene] = a_genes[other];
246  a_genes[other] = t;
247  if (m_monitorActive) {
248  a_genes[a_target_gene].setUniqueIDTemplate(a_genes[other].getUniqueID(), 1);
249  a_genes[other].setUniqueIDTemplate(a_genes[a_target_gene].getUniqueID(), 1);
250  }
251  return a_genes;
252  }
