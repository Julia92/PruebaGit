117  add(a_from_pop.getChromosome(i));

120  int canBeSelected;
121  int chromsSize = m_chromosomes.size();
122  if (chromsSize < 1) {
123  throw new IllegalStateException(
124  "Number of chromosomes must be greater 0");
125  }
126  if (a_howManyToSelect > chromsSize) {
127  canBeSelected = chromsSize;

129  else {

133  // First select all chromosomes with no fitness value computed.
134  // ------------------------------------------------------------
135  Iterator it = m_chromosomes.iterator();
136  while (it.hasNext()) {
