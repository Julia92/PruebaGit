173  protected void calcAvgFitnessHelper(int a_permutation,
174  final KeyedValues2D result) {
175  // Determine run-numbers of given permutation.
176  // -------------------------------------------
177  Map runNumbers = m_permutationRuns.get(new Integer(a_permutation));
178  if (runNumbers == null) {
179  return;
180  }
181  // Loop over all run-numbers.
182  // --------------------------
183  Iterator it = runNumbers.keySet().iterator();
184  int numRuns = runNumbers.keySet().size();
185  Integer runI;
186  while (it.hasNext()) {
187  runI = (Integer) it.next();
188  // Determine dataset of given permutation.
189  // ---------------------------------------
190  KeyedValues2D a_data = m_permutationData.get(createKey(a_permutation,
191  runI.intValue()));
192  // Determine values for current run-number and "add" them to gathered
193  // data.
194  // ------------------------------------------------------------------
195  for (int col = 0; col < a_data.getColumnCount(); col++) {
196  for (int row = 0; row < a_data.getRowCount(); row++) {
197  // Previous value (summation).
198  // --------------------------.
199  Double d = (Double) result.getValue(a_data.getRowKey(row),
200  a_data.getColumnKey(col));
201  double newValue;
202  if (d == null) {
203  newValue = 0.0d;
204  }
205  else {
206  newValue = d.doubleValue();
207  }
208  // Add current value (divided by total number of runs to get an
209  // averaged value).
210  // ------------------------------------------------------------
211  newValue +=
212  a_data.getValue(a_data.getRowKey(row), a_data.getColumnKey(col)).
213  doubleValue() / numRuns;
214  // Set averaged value back to result dataset.
215  // ------------------------------------------
216  result.setValue(new Double(newValue), a_data.getRowKey(row),
217  a_data.getColumnKey(col));
218  }
219  }
220  }
221  }

351  public GenotypeDataAvg calcPerformance(int a_permutation) {
352  int numRuns = getNumberOfRuns(a_permutation);
353  GenotypeData data;
354  GenotypeDataAvg dataAvg = new GenotypeDataAvg();
355  dataAvg.permutation = a_permutation;
356  double sizeAvg = 0.0d;
357  double fitnessAvg = 0.0d;
358  double fitnessBest = -1.0d;
359  double fitnessBestOld = -1.0d;
360  double fitness;
361  int fitnessBestGen = -1;
362  double fitnessAvgChroms;
363  double fitnessDiversityChromsOld = -1.0d;
364  double fitnessBestDeltaAvg = 0.0d;
365  double fitnessDiversity;
366  double fitnessDiversityAvg = 0.0d;
367  int size;
368  ChromosomeData chrom;
369  for (int i = 0; i < numRuns; i++) {
370  data = retrieveGenotype(a_permutation, i);
371  if (i == 0) {
372  dataAvg.generation = data.generation;
373  }
374  // Average number of chromosomes.
375  // ------------------------------
376  sizeAvg += (double) data.size / numRuns;
377  size = data.size;
378  fitnessAvgChroms = 0.0d;
379  fitnessDiversity = 0.0d;
380  double fitnessBestLocal = -1.0d;
381  for (int j = 0; j < size; j++) {
382  chrom = data.chromosomeData[j];
383  fitness = chrom.fitnessValue;
384  // diversity of fitness values over all chromosomes
385  if (j > 0) {
386  fitnessDiversity += Math.abs(fitness - fitnessDiversityChromsOld) /
387  (size - 1);
388  }
389  fitnessDiversityChromsOld = fitness;
390  // average fitness value for generation over all Chromosomes
391  fitnessAvgChroms += fitness / size;
392  // fittest chromosome in generation over all runs
393  if (fitnessBest < fitness) {
394  fitnessBest = fitness;
395  // memorize generation number in which fittest chromosome appeared
396  fitnessBestGen = data.generation;
397  }
398  // fittest chromosome in generation over current runs
399  if (fitnessBestLocal < fitness) {
400  fitnessBestLocal = fitness;
401  }
402  }
403  // average fitness value for generation over all runs
404  fitnessAvg += fitnessAvgChroms / numRuns;
405  // average fitness delta value for generation over all runs
406  fitnessDiversityAvg += fitnessDiversity / numRuns;
407  // absolute delta between two adjacent best fitness values
408  if (i > 0) {
409  fitnessBestDeltaAvg += Math.abs(fitnessBestLocal - fitnessBestOld) /
410  (numRuns - 1);
411  }
412  fitnessBestOld = fitnessBestLocal;
413  }
414  dataAvg.sizeAvg = sizeAvg;
415  dataAvg.avgFitnessValue = fitnessAvg;
416  dataAvg.bestFitnessValue = fitnessBest;
417  dataAvg.bestFitnessValueGeneration = fitnessBestGen;
418  dataAvg.avgDiversityFitnessValue = fitnessDiversityAvg;
419  dataAvg.avgBestDeltaFitnessValue = fitnessBestDeltaAvg;
420  // Store computed (averaged) data.
421  // -------------------------------
422  m_genotypeDataAvg.add(dataAvg);
423  return dataAvg;
424  }
