38  public double evaluate(IChromosome a_subject) {
39  boolean defaultComparation = a_subject.getConfiguration().
40  getFitnessEvaluator().isFitter(2, 1);
41 
42  int changeAmount = amountOfChange(a_subject);
43  int totalCoins = getTotalNumberOfCoins(a_subject);
44  int changeDifference = Math.abs(m_targetAmount - changeAmount);
45  double fitness;
46  if (defaultComparation) {
47  fitness = 0.0d;
48  }
49  else {
50  fitness = MAX_BOUND/2;
51  }
52  // Step 1: Determine distance of amount represented by solution from
53  // the target amount. If the change difference is greater than zero we
54  // will divide one by the difference in change between the
55  // solution amount and the target amount. That will give the desired
56  // effect of returning higher values for amounts closer to the target
57  // amount and lower values for amounts further away from the target
58  // amount.
59  // In the case where the change difference is zero it means that we have
60  // the correct amount and we assign a higher fitness value.
61  // ---------------------------------------------------------------------
62  if (defaultComparation) {
63  fitness += changeDifferenceBonus(MAX_BOUND/2, changeDifference);
64  }
65  else {
66  fitness -= changeDifferenceBonus(MAX_BOUND/2, changeDifference);
67  }
68  // Step 2: We divide the fitness value by a penalty based on the number of
69  // coins. The higher the number of coins the higher the penalty and the
70  // smaller the fitness value.
71  // And inversely the smaller number of coins in the solution the higher
72  // the resulting fitness value.
73  // -----------------------------------------------------------------------
74  if (defaultComparation) {
75  fitness -= computeCoinNumberPenalty(MAX_BOUND/2, totalCoins);
76  }
77  else {
78  fitness += computeCoinNumberPenalty(MAX_BOUND/2, totalCoins);
79  }
80  // Make sure fitness value is always positive.
81  // -------------------------------------------
82  return Math.max(1.0d, fitness);
83  }

85  protected double changeDifferenceBonus(double a_maxFitness,
86  int a_changeDifference) {
87  if (a_changeDifference == 0) {
88  return a_maxFitness;
89  }
90  else {
91  // we arbitrarily work with half of the maximum fitness as basis for non-
92  // optimal solutions (concerning change difference)
93  if (a_changeDifference * a_changeDifference >= a_maxFitness / 2) {
94  return 0.0d;
95  }
96  else {
97  return a_maxFitness / 2 - a_changeDifference * a_changeDifference;
98  }
99  }
100  }
101 
102  protected double computeCoinNumberPenalty(double a_maxFitness, int a_coins) {
103  if (a_coins == 1) {
104  // we know the solution cannot have less than one coin
105  return 0;
106  }
107  else {
108  // The more coins the more penalty, but not more than the maximum fitness
109  // value possible. Let's avoid linear behavior and use
110  // exponential penalty calculation instead
111  return (Math.min(a_maxFitness, a_coins * a_coins));
112  }
113  }
