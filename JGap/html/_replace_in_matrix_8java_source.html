105  public int execute_int(ProgramChromosome c, int n, Object[] args) {
106  int index;
107  // Child = index --> only relevant if not replacement for whole matrix.
108  // --------------------------------------------------------------------
109  if (m_mode != ReplacementMode.MATRIX) {
110  index = c.execute_int(n, 0, args);
111  } else {
112  index = 0;
113  }
114  // Retrieve the matrix.
115  // --------------------
116  char[][] matrix = getGPConfiguration().getMatrix(m_matrixName);
117  int count = 0;
118  if (matrix != null) {
119  int cols = matrix.length;
120  if(index >= cols) {
121  index = cols - 1;
122  }
123  else if (index < 0) {
124  index = 0;
125  }
126 
127  int rows = matrix[index].length;
128  switch (m_mode) {
129  case ROW:
130  for (int col = 0; col < cols; col++) {
131  replace(matrix[col][index]);
132  }
133  break;
134  case COLUMN:
135  for (int row = 0; row < rows; row++) {
136  replace(matrix[index][row]);
137  }
138  break;
139  case MATRIX:
140  for (int col = 0; col < cols; col++) {
141  for (int row = 0; row < rows; row++) {
142  replace(matrix[col][row]);
143  }
144  }
145  break;
146  case DIAGONAL:
147  if (cols != rows) {
148  throw new IllegalArgumentException("Cannot count diagonal of the" + " matrix, as the matrix is not square");
149  }
150  // Index less than half the number of columns/rows: first diagonal
151  // Else: second diagonal
152  if (index < cols / 2) {
153  for (int cell = 0; cell < cols; cell++) {
154  replace(matrix[cell][cell]);
155  }
156  } else {
157  for (int cell = cols - 1; cell > 0; cell--) {
158  replace(matrix[cell][cell]);
159  }
160  }
161  break;
162  }
163  }
164  return count;
165  }
