730  private int doTestEvolve_5(Configuration config)
731  throws Exception {
732  Genotype genotype = Genotype.randomInitialGenotype(config);
733  genotype.evolve(2);
734  // Reset counter. Because of static state holder we only need to do this
735  // for one chromosome referencing the same global state holder as well as
736  // all other chromosomes (of class ChromosomeForTest) do.
737  ChromosomeForTesting chrom = (ChromosomeForTesting) genotype.getPopulation().
738  getChromosome(0);
739  chrom.resetComputedTimes();
740  // Mark any chromosome as original (that is not cloned)
741  for (int i = 0; i < genotype.getPopulation().size(); i++) {
742  chrom = (ChromosomeForTesting) genotype.getPopulation().getChromosome(i);
743  chrom.resetIsCloned();
744  }
745  // Now do the test evolution --> new fitness values must be recomputed!
746  genotype.evolve(2);
747  // Check if global state holder indicates that getFitnessValue() has been
748  // called at least once for a cloned (e.g. mutated) chromosome and that for
749  // this call the to date fitness value is initial (i.e. not set).
750  chrom = (ChromosomeForTesting) genotype.getPopulation().
