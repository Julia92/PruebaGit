111  public void operate(final Population a_population,
112  final List a_candidateChromosomes) {
113  int size = Math.min(getConfiguration().getPopulationSize(),
114  a_population.size());
115  int numCrossovers = size / 2;
116  RandomGenerator generator = getConfiguration().getRandomGenerator();
117  // For each crossover, grab two random chromosomes and do what
118  // Grefenstette et al say.
119  // --------------------------------------------------------------
120  for (int i = 0; i < numCrossovers; i++) {
121  IChromosome origChrom1 = a_population.getChromosome(generator.
122  nextInt(size));
123  IChromosome firstMate = (IChromosome)origChrom1.clone();
124  IChromosome origChrom2 = a_population.getChromosome(generator.
125  nextInt(size));
126  IChromosome secondMate = (IChromosome)origChrom2.clone();
127  // In case monitoring is active, support it.
128  // -----------------------------------------
129  if (m_monitorActive) {
130  firstMate.setUniqueIDTemplate(origChrom1.getUniqueID(), 1);
131  firstMate.setUniqueIDTemplate(origChrom2.getUniqueID(), 2);
132  secondMate.setUniqueIDTemplate(origChrom1.getUniqueID(), 1);
133  secondMate.setUniqueIDTemplate(origChrom2.getUniqueID(), 2);
134  }
135  operate(firstMate, secondMate);
136  // Add the modified chromosomes to the candidate pool so that
137  // they'll be considered for natural selection during the next
138  // phase of evolution.
139  // -----------------------------------------------------------
140  a_candidateChromosomes.add(firstMate);
141  a_candidateChromosomes.add(secondMate);
142  }
143  }

176  protected Gene[] operate(final Gene[] a_g1, final Gene[] a_g2) {
177  int n = a_g1.length;
178  LinkedList out = new LinkedList();
179  TreeSet not_picked = new TreeSet();
180  out.add(a_g1[m_startOffset]);
181  for (int j = m_startOffset + 1; j < n; j++) { // g[m_startOffset] picked
182  if (ASSERTIONS && not_picked.contains(a_g1[j])) {
183  throw new Error("All genes must be different for "
184  + getClass().getName()
185  + ". The gene " + a_g1[j] + "[" + j
186  + "] occurs more "
187  + "than once in one of the chromosomes. ");
188  }
189  not_picked.add(a_g1[j]);
190  }
191  if (ASSERTIONS) {
192  if (a_g1.length != a_g2.length) {
193  throw new Error("Chromosome sizes must be equal");
194  }
195  for (int j = m_startOffset; j < n; j++) {
196  if (!not_picked.contains(a_g2[j])) {
197  if (!a_g1[m_startOffset].equals(a_g2[j])) {
198  throw new Error("Chromosome gene sets must be identical."
199  + " First gene set: " + a_g1
200  + ", second gene set: " + a_g2);
201  }
202  }
203  }
204  }
205  while (not_picked.size() > 1) {
206  Gene last = (Gene) out.getLast();
207  Gene n1 = findNext(a_g1, last);
208  Gene n2 = findNext(a_g2, last);
209  Gene picked, other;
210  boolean pick1;
211  if (n1 == null) {
212  pick1 = false;
213  }
214  else if (n2 == null) {
215  pick1 = true;
216  }
217  else {
218  pick1 = distance(last, n1) < distance(last, n2);
219  }
220  if (pick1) {
221  picked = n1;
222  other = n2;
223  }
224  else {
225  picked = n2;
226  other = n1;
227  }
228  if (out.contains(picked)) {
229  picked = other;
230  }
231  if (picked == null || out /* still */.contains(picked)) {
232  // select a non-selected // it is not random
233  picked = (Gene) not_picked.first();
234  }
235  out.add(picked);
236  not_picked.remove(picked);
237  }
238  if (ASSERTIONS && not_picked.size() != 1) {
239  throw new Error("Given Gene not correctly created (must have length > 1"
240  + ")");
241  }
242  out.add(not_picked.last());
243  Gene[] g = new Gene[n];
244  Iterator gi = out.iterator();
245  for (int i = 0; i < m_startOffset; i++) {
246  g[i] = a_g1[i];
247  }
248  if (ASSERTIONS) {
249  if (out.size() != g.length - m_startOffset) {
250  throw new Error("Unexpected internal error. "
251  + "These two must be equal: " + out.size()
252  + " and " + (g.length - m_startOffset) + ", g.length "
253  + g.length + ", start offset " + m_startOffset);
254  }
255  }
256  for (int i = m_startOffset; i < g.length; i++) {
257  g[i] = (Gene) gi.next();
258  }
259  return g;
260  }
