1108  public synchronized void verifyStateIsValid()
1109  throws InvalidConfigurationException {
1110  // First, make sure all of the required fields have been set to
1111  // appropriate values.
1112  // ------------------------------------------------------------
1113  if (m_objectiveFunction == null && m_bulkObjectiveFunction == null) {
1114  throw new InvalidConfigurationException(
1115  "A desired fitness function or bulk fitness function must " +
1116  "be specified in the active configuration.");
1117  }
1118  if (m_sampleChromosome == null) {
1119  throw new InvalidConfigurationException(
1120  "A sample instance of the desired Chromosome " +
1121  "setup must be specified in the active configuration.");
1122  }
1123  if (m_preSelectors.size() == 0 && m_postSelectors.size() == 0) {
1124  throw new InvalidConfigurationException(
1125  "At least one desired natural selector must be specified in the"
1126  + " active configuration.");
1127  }
1128  if (m_randomGenerator == null) {
1129  throw new InvalidConfigurationException(
1130  "A desired random number generator must be specified in the " +
1131  "active configuration.");
1132  }
1133  if (m_eventManager == null) {
1134  throw new InvalidConfigurationException(
1135  "A desired event manager must be specified in the active " +
1136  "configuration.");
1137  }
1138  if (m_geneticOperators.isEmpty()) {
1139  throw new InvalidConfigurationException(
1140  "At least one genetic operator must be specified in the " +
1141  "configuration.");
1142  }
1143  if (m_chromosomeSize <= 0) {
1144  throw new InvalidConfigurationException(
1145  "A chromosome size greater than zero must be specified in " +
1146  "the active configuration.");
1147  }
1148  if (m_config.m_populationSize <= 0) {
1149  throw new InvalidConfigurationException(
1150  "A population size greater than zero must be specified in " +
1151  "the active configuration.");
1152  }
1153  if (m_fitnessEvaluator == null) {
1154  throw new IllegalArgumentException(
1155  "The fitness evaluator must not be null.");
1156  }
1157  // Next, it's critical that each Gene implementation in the sample
1158  // Chromosome has a working equals() method, or else the genetic
1159  // engine will end up failing in mysterious and unpredictable ways.
1160  // We therefore verify right here that this method is working properly
1161  // in each of the Gene implementations used in the sample Chromosome.
1162  // -------------------------------------------------------------------
1163  Gene[] sampleGenes = m_sampleChromosome.getGenes();
1164  for (int i = 0; i < sampleGenes.length; i++) {
1165  Gene sampleCopy = sampleGenes[i].newGene();
1166  sampleCopy.setAllele(sampleGenes[i].getAllele());
1167  if (! (sampleCopy.equals(sampleGenes[i]))) {
1168  throw new InvalidConfigurationException(
1169  "The sample Gene at gene position (locus) "
1170  + i
1171  + " does not appear to have a working equals() or compareTo()"
1172  + " method.\n"
1173  + "It could also be that you forgot to implement method"
1174  + " newGene() in your Gene implementation.\n"
1175  + "When tested, the method returned false when comparing "
1176  + "the sample gene with a gene of the same type and "
1177  + "possessing the same value (allele).");
1178  }
1179  }
1180  }
