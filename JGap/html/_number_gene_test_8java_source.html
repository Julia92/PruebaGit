519  protected void calculateBoundsUnitsToIntegerUnitsRatio() {
520  int divisor = m_upperBounds - m_lowerBounds + 1;
521  if (divisor == 0) {
522  m_boundsUnitsToIntegerUnits = INTEGER_RANGE;

529  protected void mapValueToWithinBounds() {
530  if (getAllele() != null) {
531  Integer i_value = ( (Integer) getAllele());
532  // If the value exceeds either the upper or lower bounds, then
533  // map the value to within the legal range. To do this, we basically
534  // calculate the distance between the value and the integer min,
535  // determine how many bounds units that represents, and then add
536  // that number of units to the upper bound.
537  // -----------------------------------------------------------------
538  if (i_value.intValue() > m_upperBounds
539  || i_value.intValue() < m_lowerBounds) {
540  long differenceFromIntMin = (long) Integer.MIN_VALUE
541  + (long) i_value.intValue();
542  int differenceFromBoundsMin = (int) (differenceFromIntMin
543  / m_boundsUnitsToIntegerUnits);
544  setAllele(new Integer(m_upperBounds + differenceFromBoundsMin));
545  }
546  }
547  }
