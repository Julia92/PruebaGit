163  public void growOrFull(int a_depth, boolean a_grow, int a_maxNodes,
164  boolean[] a_fullModeAllowed, int a_tries) {
165  GPConfiguration conf = getGPConfiguration();
166  // The number of chromosomes to create.
167  // ------------------------------------
168  int size = m_chromosomes.length;
169  for (int i = 0; i < size; i++) {
170  try {
171  // Construct a chromosome with place for a_maxNodes nodes.
172  // -------------------------------------------------------
173  m_chromosomes[i] = new ProgramChromosome(conf, a_maxNodes, this);
174  } catch (InvalidConfigurationException iex) {
175  throw new RuntimeException(iex);
176  }
177  m_chromosomes[i].setArgTypes(getArgTypes()[i]);
178  // If there are ADF's in the nodeSet, then set their type according to
179  // the chromosome it references.
180  // -------------------------------------------------------------------
181  int len = getNodeSets()[i].length;
182  for (int j = 0; j < len; j++) {
183  if (getNodeSets()[i][j] instanceof ADF) {
184  ( (ADF) getNodeSets()[i][j]).setReturnType(
185  getTypes()[ ( (ADF) getNodeSets()[i][j]).getChromosomeNum()]);
186  }
187  }
188  }
189  int depth;
190  for (int i = 0; i < size; i++) {
191  // Restrict depth to input params.
192  // -------------------------------
193  if (getMaxDepths() != null && a_depth > getMaxDepths()[i]) {
194  depth = getMaxDepths()[i];
195  }
196  else {
197  if (getMinDepths() != null && a_depth < getMinDepths()[i]) {
198  depth = getMinDepths()[i];
199  }
200  else {
201  depth = a_depth;
202  }
203  }
204  // Decide whether to use grow mode or full mode.
205  // Here, the program is finally created.
206  // ---------------------------------------------
207  if (a_grow || a_fullModeAllowed == null || !a_fullModeAllowed[i]) {
208  m_chromosomes[i].growOrFull(i, depth, getType(i), getArgType(i),
209  getNodeSet(i), true, a_tries);
210  }
211  else {
212  m_chromosomes[i].growOrFull(i, depth, getType(i), getArgType(i),
213  getNodeSet(i), false, a_tries);
214  }
215  }
216  if (getGPConfiguration().isUseProgramCache()) {
217  // Cache fitness value by checking if a program with same
218  // representation was computed before.
219  // ------------------------------------------------------
220  GPProgramInfo pcInfo = getGPConfiguration().readProgramCache(this);
221  if (pcInfo == null) {
222  pcInfo = putToCache(this);
223  }
224  else {
225  setFitnessValue(pcInfo.getFitnessValue());
226  }
227  }

284  }
285  }

287  }

499  public int compareTo(Object a_other) {
500  // First, if the other Chromosome is null, then this chromosome is
501  // automatically the "greater" Chromosome.
502  // ---------------------------------------------------------------
503  if (a_other == null) {
504  return 1;
505  }
506  int size = size();
507  GPProgram other = (GPProgram) a_other;
508  ProgramChromosome[] otherChroms = other.m_chromosomes;
509  // If the other Chromosome doesn't have the same number of genes,
510  // then whichever has more is the "greater" Chromosome.
511  // --------------------------------------------------------------
512  if (other.size() != size) {
513  return size() - other.size();
514  }
515  // Next, compare the gene values (alleles) for differences. If
516  // one of the genes is not equal, then we return the result of its
517  // comparison.
518  // ---------------------------------------------------------------
519  Arrays.sort(m_chromosomes);
520  Arrays.sort(otherChroms);
521  for (int i = 0; i < size; i++) {
522  int comparison = m_chromosomes[i].compareTo(otherChroms[i]);
523  if (comparison != 0) {
524  return comparison;
525  }
526  }
527  // Everything is equal. Return zero.
528  // ---------------------------------
529  return 0;
530  }
