198  public IChromosome findOptimalPath(final Object a_initial_data)
199  throws Exception {
200  m_config = createConfiguration(a_initial_data);
201  FitnessFunction myFunc = createFitnessFunction(a_initial_data);
202  m_config.setFitnessFunction(myFunc);
203  // Now we need to tell the Configuration object how we want our
204  // Chromosomes to be setup. We do that by actually creating a
205  // sample Chromosome and then setting it on the Configuration
206  // object.
207  // --------------------------------------------------------------
208  IChromosome sampleChromosome = createSampleChromosome(a_initial_data);
209  m_config.setSampleChromosome(sampleChromosome);
210  // Finally, we need to tell the Configuration object how many
211  // Chromosomes we want in our population. The more Chromosomes,
212  // the larger number of potential solutions (which is good for
213  // finding the answer), but the longer it will take to evolve
214  // the population (which could be seen as bad). We'll just set
215  // the population size to 500 here.
216  // ------------------------------------------------------------
217  m_config.setPopulationSize(getPopulationSize());
218  // Create random initial population of Chromosomes.
219  // ------------------------------------------------
220 
221  // As we cannot allow the normal mutations if this task,
222  // we need multiple calls to createSampleChromosome.
223  // -----------------------------------------------------
224  IChromosome[] chromosomes =
225  new IChromosome[m_config.getPopulationSize()];
226  Gene[] samplegenes = sampleChromosome.getGenes();
227  for (int i = 0; i < chromosomes.length; i++) {
228  Gene[] genes = new Gene[samplegenes.length];
229  for (int k = 0; k < genes.length; k++) {
230  genes[k] = samplegenes[k].newGene();
231  genes[k].setAllele(samplegenes[k].getAllele());
232  }
233  chromosomes[i] = new Chromosome(m_config, genes);
234  }
235  // Create the genotype. We cannot use Genotype.randomInitialGenotype,
236  // Because we need unique gene values (representing the indices of the
237  // cities of our problem).
238  // -------------------------------------------------------------------
239  Genotype population = new Genotype(m_config,
240  new Population(m_config, chromosomes));
241  IChromosome best = null;
242  // Evolve the population. Since we don't know what the best answer
243  // is going to be, we just evolve the max number of times.
244  // ---------------------------------------------------------------
245  Evolution:
246  for (int i = 0; i < getMaxEvolution(); i++) {
247  population.evolve();
248  best = population.getFittestChromosome();
249  }
250  // Return the best solution we found.
251  // ----------------------------------
252  return best;
253  }
