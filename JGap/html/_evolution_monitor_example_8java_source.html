47  public static void makeChangeForAmount(int a_targetChangeAmount)
48  throws Exception {
49  // Start with a DefaultConfiguration, which comes setup with the
50  // most common settings.
51  // -------------------------------------------------------------
52  Configuration conf = new DefaultConfiguration();
53  // Care that the fittest individual of the current population is
54  // always taken to the next generation.
55  // Consider: With that, the pop. size may exceed its original
56  // size by one sometimes!
57  // -------------------------------------------------------------
58  conf.setPreservFittestIndividual(true);
59  // Set the fitness function we want to use, which is our
60  // MinimizingMakeChangeFitnessFunction. We construct it with
61  // the target amount of change passed in to this method.
62  // ---------------------------------------------------------
63  FitnessFunction myFunc =
64  new MinimizingMakeChangeFitnessFunction(a_targetChangeAmount);
65  conf.setFitnessFunction(myFunc);
66  // Now we need to tell the Configuration object how we want our
67  // Chromosomes to be setup. We do that by actually creating a
68  // sample Chromosome and then setting it on the Configuration
69  // object. As mentioned earlier, we want our Chromosomes to each
70  // have four genes, one for each of the coin types. We want the
71  // values (alleles) of those genes to be integers, which represent
72  // how many coins of that type we have. We therefore use the
73  // IntegerGene class to represent each of the genes. That class
74  // also lets us specify a lower and upper bound, which we set
75  // to sensible values for each coin type.
76  // --------------------------------------------------------------
77  Gene[] sampleGenes = new Gene[4];
78  sampleGenes[0] = new IntegerGene(conf, 0, 3 * 10); // Quarters
79  sampleGenes[1] = new IntegerGene(conf, 0, 2 * 10); // Dimes
80  sampleGenes[2] = new IntegerGene(conf, 0, 1 * 10); // Nickels
81  sampleGenes[3] = new IntegerGene(conf, 0, 4 * 10); // Pennies
82  IChromosome sampleChromosome = new Chromosome(conf, sampleGenes);
83  conf.setSampleChromosome(sampleChromosome);
84  // Finally, we need to tell the Configuration object how many
85  // Chromosomes we want in our population. The more Chromosomes,
86  // the larger number of potential solutions (which is good for
87  // finding the answer), but the longer it will take to evolve
88  // the population (which could be seen as bad).
89  // ------------------------------------------------------------
90  conf.setPopulationSize(80);
91  // Create random initial population of Chromosomes.
92  // ------------------------------------------------
93  Genotype population = Genotype.randomInitialGenotype(conf);
94  long startTime = System.currentTimeMillis();
95  // Configure monitors to let the evolution stop when defined criteria
96  // are met.
97  // ------------------------------------------------------------------
98  List monitors = new Vector();
99  monitors.add(new TimedMonitor(6));
100  monitors.add(new FitnessImprovementMonitor(1, 3, 5.0d));
101  IEvolutionMonitor monitor = new ChainedMonitors(monitors, 2);
102  // Start the evolution.
103  // --------------------
104  List<String> messages = population.evolve(monitor);
105  if (messages.size() > 0) {
106  for (String msg : messages) {
107  System.out.println("Message from monitor: " + msg+"\n");
108  }
109  }
110  // Evaluate the result.
111  // --------------------
112  long endTime = System.currentTimeMillis();
113  System.out.println("Total evolution time: " + (endTime - startTime)
114  + " ms");
115  // Display the best solution we found.
116  // -----------------------------------
117  IChromosome bestSolutionSoFar = population.getFittestChromosome();
118  System.out.println("The best solution has a fitness value of " +
119  bestSolutionSoFar.getFitnessValue());
120  System.out.println("It contained the following: ");
121  System.out.println("\t" +
122  MinimizingMakeChangeFitnessFunction.
123  getNumberOfCoinsAtGene(
124  bestSolutionSoFar, 0) + " quarters.");
125  System.out.println("\t" +
126  MinimizingMakeChangeFitnessFunction.
127  getNumberOfCoinsAtGene(
128  bestSolutionSoFar, 1) + " dimes.");
129  System.out.println("\t" +
130  MinimizingMakeChangeFitnessFunction.
131  getNumberOfCoinsAtGene(
132  bestSolutionSoFar, 2) + " nickels.");
133  System.out.println("\t" +
134  MinimizingMakeChangeFitnessFunction.
135  getNumberOfCoinsAtGene(
136  bestSolutionSoFar, 3) + " pennies.");
137  System.out.println("For a total of " +
138  MinimizingMakeChangeFitnessFunction.amountOfChange(
139  bestSolutionSoFar) + " cents in " +
140  MinimizingMakeChangeFitnessFunction.
141  getTotalNumberOfCoins(
142  bestSolutionSoFar) + " coins.");
143  }
