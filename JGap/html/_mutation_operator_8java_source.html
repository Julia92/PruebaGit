136  public void operate(final Population a_population,
137  final List a_candidateChromosomes) {
138  if (a_population == null || a_candidateChromosomes == null) {
139  // Population or candidate chromosomes list empty:
140  // nothing to do.
141  // -----------------------------------------------
142  return;
143  }
144  if (m_config.m_mutationRate == 0 && m_mutationRateCalc == null) {
145  // If the mutation rate is set to zero and dynamic mutation rate is
146  // disabled, then we don't perform any mutation.
147  // ----------------------------------------------------------------
148  return;
149  }
150  // Determine the mutation rate. If dynamic rate is enabled, then
151  // calculate it using the IUniversalRateCalculator instance.
152  // Otherwise, go with the mutation rate set upon construction.
153  // -------------------------------------------------------------
154  boolean mutate = false;
155  RandomGenerator generator = getConfiguration().getRandomGenerator();
156  // It would be inefficient to create copies of each Chromosome just
157  // to decide whether to mutate them. Instead, we only make a copy
158  // once we've positively decided to perform a mutation.
159  // ----------------------------------------------------------------
160  int size = Math.min(getConfiguration().getPopulationSize(),
161  a_population.size());
162  IGeneticOperatorConstraint constraint = getConfiguration().
163  getJGAPFactory().getGeneticOperatorConstraint();
164  //
165  for (int i = 0; i < size; i++) {
166  IChromosome chrom = a_population.getChromosome(i);
167  Gene[] genes1 = chrom.getGenes();
168  IChromosome copyOfChromosome = null;
169  Gene[] genes = null;
170  // For each Chromosome in the population...
171  // ----------------------------------------
172  for (int j = 0; j < genes1.length; j++) {
173  if (m_mutationRateCalc != null) {
174  // If it's a dynamic mutation rate then let the calculator decide
175  // whether the current gene should be mutated.
176  // --------------------------------------------------------------
177  mutate = m_mutationRateCalc.toBePermutated(chrom, j);
178  }
179  else {
180  // Non-dynamic, so just mutate based on the the current rate.
181  // In fact we use a rate of 1/m_mutationRate.
182  // ----------------------------------------------------------
183  mutate = (generator.nextInt(m_config.m_mutationRate) == 0);
184  }
185  if (mutate) {
186  // Verify that crossover allowed.
187  // ------------------------------
189  if (constraint != null) {
190  List v = new Vector();
191  v.add(chrom);
192  if (!constraint.isValid(a_population, v, this)) {
193  continue;
194  }
195  }
196  // Now that we want to actually modify the Chromosome,
197  // let's make a copy of it (if we haven't already) and
198  // add it to the candidate chromosomes so that it will
199  // be considered for natural selection during the next
200  // phase of evolution. Then we'll set the gene's value
201  // to a random value as the implementation of our
202  // "mutation" of the gene.
203  // ---------------------------------------------------
204  if (copyOfChromosome == null) {
205  // ...take a copy of it...
206  // -----------------------
207  copyOfChromosome = (IChromosome) chrom.clone();
208  // ...add it to the candidate pool...
209  // ----------------------------------
210  a_candidateChromosomes.add(copyOfChromosome);
211  // ...then mutate all its genes...
212  // -------------------------------
213  genes = copyOfChromosome.getGenes();
214  // In case monitoring is active, support it.
215  // -----------------------------------------
216  if (m_monitorActive) {
217  copyOfChromosome.setUniqueIDTemplate(chrom.getUniqueID(), 1);
218  }
219  }
220  // Process all atomic elements in the gene. For a StringGene this
221  // would be as many elements as the string is long , for an
222  // IntegerGene, it is always one element.
223  // --------------------------------------------------------------
224  if (genes[j] instanceof ICompositeGene) {
225  ICompositeGene compositeGene = (ICompositeGene) genes[j];
226  if (m_monitorActive) {
227  compositeGene.setUniqueIDTemplate(chrom.getGene(j).getUniqueID(), 1);
228  }
229  for (int k = 0; k < compositeGene.size(); k++) {
230  mutateGene(compositeGene.geneAt(k), generator);
231  if (m_monitorActive) {
232  compositeGene.geneAt(k).setUniqueIDTemplate(
233  ( (ICompositeGene) chrom.getGene(j)).geneAt(k).getUniqueID(),
234  1);
235  }
236  }
237  }
238  else {
239  mutateGene(genes[j], generator);
240  if (m_monitorActive) {
241  genes[j].setUniqueIDTemplate(chrom.getGene(j).getUniqueID(), 1);
242  }
243  }
244  }
245  }
246  }
247  }

258  private void mutateGene(final Gene a_gene, final RandomGenerator a_generator) {
259  for (int k = 0; k < a_gene.size(); k++) {
260  // Retrieve value between 0 and 1 (not included) from generator.
261  // Then map this value to range -1 and 1 (-1 included, 1 not).
262  // -------------------------------------------------------------
263  double percentage = -1 + a_generator.nextDouble() * 2;
264  // Mutate atomic element by calculated percentage.
265  // -----------------------------------------------
266  a_gene.applyMutation(k, percentage);
267  }
268  }
