141  public void selectChromosomes(final int a_howManyToSelect,
142  Population a_to_pop) {
143  int canBeSelected;
144  int chromsSize = m_chromosomes.size();
145  if (a_howManyToSelect > chromsSize) {
146  canBeSelected = chromsSize;
147  }
148  else {
149  canBeSelected = a_howManyToSelect;
150  }
151  int neededSize = a_howManyToSelect;
152  double origRate;
153  origRate = m_config.m_originalRate;
154  if (origRate < 1.0d) {
155  canBeSelected = (int) Math.round( (double) canBeSelected *
156  origRate);
157  if (canBeSelected < 1) {
158  canBeSelected = 1;
159  }
160  }
161  // Sort the collection of chromosomes previously added for evaluation.
162  // Only do this if necessary.
163  // -------------------------------------------------------------------
164  if (m_needsSorting) {
165  Collections.sort(m_chromosomes.getChromosomes(),
166  m_fitnessValueComparator);
167  m_needsSorting = false;
168  }
169  // To select a chromosome, we just go thru the sorted list.
170  // --------------------------------------------------------
171  IChromosome selectedChromosome;
172  for (int i = 0; i < canBeSelected; i++) {
173  selectedChromosome = m_chromosomes.getChromosome(i);
174  selectedChromosome.setIsSelectedForNextGeneration(true);
175  a_to_pop.addChromosome(selectedChromosome);
176  }
177  if (getDoubletteChromosomesAllowed()) {
178  int toAdd;
179  toAdd = neededSize - a_to_pop.size();
180  // Add existing Chromosome's to fill up the return
181  // result to contain the desired number of Chromosome's.
182  // -----------------------------------------------------
183  for (int i = 0; i < toAdd; i++) {
184  selectedChromosome = m_chromosomes.getChromosome(i % chromsSize);
185  ICloneHandler cloner = getConfiguration().getJGAPFactory().
186  getCloneHandlerFor(selectedChromosome, null);
187  IChromosome cloned = null;
188  if (cloner != null) {
189  try {
190  int age = selectedChromosome.getAge() + 1;
191  cloned = (IChromosome) cloner.perform(
192  selectedChromosome, null, null);
193  cloned.setAge(age);
194  cloned.setIsSelectedForNextGeneration(true);
195  if(m_monitorActive) {
196  cloned.setUniqueIDTemplate(selectedChromosome.getUniqueID(), 1);
197  }
198  } catch (Exception ex) {
199  ex.printStackTrace();
200  }
201  }
202  if(cloned != null) {
203  selectedChromosome = cloned;
204  }
205  a_to_pop.addChromosome(selectedChromosome);
206  }
207  }
