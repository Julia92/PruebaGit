816  public int compareTo(Object other) {
817  // First, if the other Chromosome is null, then this chromosome is
818  // automatically the "greater" Chromosome.
819  // ---------------------------------------------------------------
820  if (other == null) {
821  return 1;
822  }
823  int size = size();
824  IChromosome otherChromosome = (IChromosome) other;
825  Gene[] otherGenes = otherChromosome.getGenes();
826  // If the other Chromosome doesn't have the same number of genes,
827  // then whichever has more is the "greater" Chromosome.
828  // --------------------------------------------------------------
829  if (otherChromosome.size() != size) {
830  return size() - otherChromosome.size();
831  }
832  // Next, compare the gene values (alleles) for differences. If
833  // one of the genes is not equal, then we return the result of its
834  // comparison.
835  // ---------------------------------------------------------------
836  for (int i = 0; i < size; i++) {
837  int comparison = getGene(i).compareTo(otherGenes[i]);
838  if (comparison != 0) {
839  return comparison;
840  }
841  }
842  // Compare current fitness value.
843  // ------------------------------
844  if (m_fitnessValue != otherChromosome.getFitnessValueDirectly()) {
845  FitnessEvaluator eval = getConfiguration().getFitnessEvaluator();
846  if (eval != null) {
847  if (eval.isFitter(m_fitnessValue,
848  otherChromosome.getFitnessValueDirectly())) {
849  return 1;
850  }
851  else {
852  return -1;
853  }
854  }
855  else {
856  // undetermined order, but unequal!
857  // --------------------------------
858  return -1;
859  }
860  }
861  if (m_compareAppData) {
862  // Compare application data.
863  // -------------------------
864  if (getApplicationData() == null) {
865  if (otherChromosome.getApplicationData() != null) {
866  return -1;
867  }
868  }
869  else if (otherChromosome.getApplicationData() == null) {
870  return 1;
871  }
872  else {
873  if (getApplicationData() instanceof Comparable) {
874  try {
875  return ( (Comparable) getApplicationData()).compareTo(
876  otherChromosome.getApplicationData());
877  } catch (ClassCastException cex) {
879  return -1;
880  }
881  }
882  else {
883  return getApplicationData().getClass().getName().compareTo(
884  otherChromosome.getApplicationData().getClass().getName());
885  }
886  }
887  }

889  // ---------------------------------
890  return 0;
