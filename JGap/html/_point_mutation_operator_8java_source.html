38  // nothing to do.
39  // -----------------------------------------------
40  return;
41  }
42  if (getMutationRate() == 0 && getMutationRateCalc() == null) {
43  // If the mutation rate is set to zero and dynamic mutation rate is
44  // disabled, then we don't perform any mutation.
45  // ----------------------------------------------------------------
46  return;
47  }
48  GAConfiguration conf = (GAConfiguration) getConfiguration();
49  // Determine the mutation rate. If dynamic rate is enabled, then
50  // calculate it using the IUniversalRateCalculator instance.
51  // Otherwise, go with the mutation rate set upon construction.
52  // -------------------------------------------------------------
53  boolean mutate = false;
54  RandomGenerator generator = conf.getRandomGenerator();
55  // It would be inefficient to create copies of each Chromosome just
56  // to decide whether to mutate them. Instead, we only make a copy
57  // once we've positively decided to perform a mutation.
58  // ----------------------------------------------------------------
59  int size = Math.min(conf.getPopulationSize(),
60  a_population.size());
61  IGeneticOperatorConstraint constraint = conf.getJGAPFactory().
62  getGeneticOperatorConstraint();
63  for (int i = 0; i < size; i++) {
64  IChromosome chrom = a_population.getChromosome(i);
65  Gene[] genes = chrom.getGenes();
66  IChromosome copyOfChromosome = null;
67  // For each Chromosome in the population...
68  // ----------------------------------------
69 
70  /* Find a polygon to mutate */
71  int polygon = generator.nextInt(conf.getMaxPolygons()) *
72  GAInitialChromosomeFactory.getNumberOfGenesPerPolygon();
73  int point = generator.nextInt(GAInitialChromosomeFactory.POINTS) *
74  GAInitialChromosomeFactory.getNumberOfGenesPerPoint();
75  for (int c = 0; c < GAInitialChromosomeFactory.getNumberOfGenesPerPoint();

78  GAInitialChromosomeFactory.getNumberOfColorGenesPerPolygon();

80  // If it's a dynamic mutation rate then let the calculator decide
81  // whether the current gene should be mutated.
82  // --------------------------------------------------------------
83  mutate = getMutationRateCalc().toBePermutated(chrom, target);
84  }
85  else {

87  // In fact we use a rate of 1/m_mutationRate.
88  // ----------------------------------------------------------
89  mutate = (generator.nextInt(getMutationRate()) == 0);

93  // ------------------------------
95  if (constraint != null) {

98  if (!constraint.isValid(a_population, v, this)) {
99  continue;
100  }
101  }

103  // let's make a copy of it (if we haven't already) and
104  // add it to the candidate chromosomes so that it will
105  // be considered for natural selection during the next
106  // phase of evolution. Then we'll set the gene's value
107  // to a random value as the implementation of our
108  // "mutation" of the gene.
109  // ---------------------------------------------------

111  // ...take a copy of it...
112  // -----------------------

115  // ----------------------------------

118  // -------------------------------

121  // -----------------------------------------

123  copyOfChromosome.setUniqueIDTemplate(chrom.getUniqueID(), 1);
124  }
125  }

127  // would be as many elements as the string is long , for an
128  // IntegerGene, it is always one element.
129  // --------------------------------------------------------------

132  if (m_monitorActive) {

134  }
135  for (int k = 0; k < compositeGene.size(); k++) {
136  mutateGene(compositeGene.geneAt(k), generator);
137  if (m_monitorActive) {
138  compositeGene.geneAt(k).setUniqueIDTemplate(
139  ( (ICompositeGene) chrom.getGene(target)).geneAt(k).getUniqueID(),
140  1);
141  }
142  }
143  }
144  else {
145  mutateGene(genes[target], generator);
146  if (m_monitorActive) {
147  genes[target].setUniqueIDTemplate(chrom.getGene(target).getUniqueID(), 1);
148  }
149  }
150  }
151  }

155  private void mutateGene(final Gene a_gene, final RandomGenerator a_generator) {
156  for (int k = 0; k < a_gene.size(); k++) {
157  // Retrieve value between 0 and 1 (not included) from generator.
158  // Then map this value to range -1 and 1 (-1 included, 1 not).
159  // -------------------------------------------------------------

161  // Mutate atomic element by calculated percentage.
