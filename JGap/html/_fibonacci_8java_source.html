134  private static int fib_iter(int a_index) {
135  // 1
136  if (a_index == 0 || a_index == 1) {

138  }
139  // 2

146  a = b; //B: TransferMemory("mem1","mem0")

308  public double computeRawFitness(final IGPProgram a_program) {
309  double error = 0.0f;
310  Object[] noargs = new Object[0];
311  // Initialize local stores.
312  // ------------------------
313  a_program.getGPConfiguration().clearStack();
314  a_program.getGPConfiguration().clearMemory();
315  // Compute fitness for each program.
316  // ---------------------------------
318  for (int i = 2; i < NUMFIB; i++) {
319  for (int j = 0; j < a_program.size(); j++) {
320  vx.set(x[i]);
321  try {
322  try {
323  // Init. params (a_program.getTypes()) distinguish program flow.
324  // This could be coded dynamically but that would slow down
325  // things a lot.
326  // -------------------------------------------------------------
327  if (j == a_program.size() - 1) {
328  // Only evaluate after whole GP program was run.
329  // ---------------------------------------------
330  double result = a_program.execute_int(j, noargs);
331  error += Math.abs(result - y[i]);
332  }
333  else {
334  // Execute memory manipulating subprograms.
335  // ----------------------------------------
336  a_program.execute_void(j, noargs);
337  }
338  } catch (IllegalStateException iex) {
339  error = GPFitnessFunction.MAX_FITNESS_VALUE;
340  break;
341  }
342  } catch (ArithmeticException ex) {
343  System.out.println("x = " + x[i].intValue());
344  System.out.println(a_program.getChromosome(j));
345  throw ex;
346  }
347  }
348  }
349  if (a_program.getGPConfiguration().stackSize() > 0) {
350  error = GPFitnessFunction.MAX_FITNESS_VALUE;
351  }
352  if (error < 0.000001) {
353  error = 0.0d;
354  }
355  else if (error < GPFitnessFunction.MAX_FITNESS_VALUE) {
357  }
358  return error;
359  }
