137  private int[][] readMap(String a_filename)
138  throws Exception {
139  LineNumberReader lnr;
140  try {
141  lnr = new LineNumberReader(new FileReader(a_filename));
142  } catch (FileNotFoundException fex) {
143  throw new FileNotFoundException("File not found: " +
144  new File(".").getAbsolutePath() +
145  a_filename);
146  }
147  // Read dimensions of trail and the number of ants.
148  try {
149  StringTokenizer st = new StringTokenizer(lnr.readLine());
150  m_maxx = Integer.parseInt(st.nextToken());
151  m_maxy = Integer.parseInt(st.nextToken());
152  m_popSize = Integer.parseInt(st.nextToken());
153  int[][] result = new int[m_maxx][m_maxy];
154  m_ants = new Ant[m_popSize];
155  int y;
156  int antIndex = 0;
157  for (y = 0; y < m_maxy; y++) {
158  String s = lnr.readLine();
159  System.out.println(s);
160  if (s == null) {
161  throw new RuntimeException("Ant map file ended prematurely");
162  }
163  int x;
164  for (x = 0; x < s.length() & x < m_maxx; x++) {
165  if (s.charAt(x) == ' ') {
166  result[x][y] = AntMap.EMPTY;
167  }
168  else if (s.charAt(x) == 'b') {
169  result[x][y] = AntMap.BLACK;
170  }
171  else if (s.charAt(x) == 'g') {
172  result[x][y] = AntMap.GRAY;
173  }
174  else if (s.charAt(x) == 's') {
175  result[x][y] = AntMap.STRIPED;
176  }
177  else if (s.charAt(x) == 'A') {
178  result[x][y] = AntMap.EMPTY;
179  m_ants[antIndex] = new Ant(x, y);
180  antIndex++;
181  }
182  else if (s.charAt(x) == 'S') {
183  result[x][y] = AntMap.STRIPED;
184  m_ants[antIndex] = new Ant(x, y);
185  antIndex++;
186  }
187  else if (s.charAt(x) == 'G') {
188  result[x][y] = AntMap.GRAY;
189  m_ants[antIndex] = new Ant(x, y);
190  antIndex++;
191  }
192  else if (s.charAt(x) == 'B') {
193  result[x][y] = AntMap.BLACK;
194  m_ants[antIndex] = new Ant(x, y);
195  antIndex++;
196  }
197  else {
198  throw new RuntimeException("Bad character '" + s.charAt(x) +
199  "' on line number " + lnr.getLineNumber() +
200  " of the Ant map file.");
201  }
202  }
203  // fill out rest of X's
204  for (int z = x; z < m_maxx; z++) {
205  result[z][y] = AntMap.EMPTY;
206  }
207  }
208  // fill out rest of Y's
209  for (int z = y; z < m_maxy; z++) {
210  for (int x = 0; x < m_maxx; x++) {
211  result[x][z] = AntMap.EMPTY;
212  }
213  }
214  return result;
215  } catch (NumberFormatException e) {
216  throw new RuntimeException(
217  "The Ant map file does not begin with x and y and nbrOfAnts integer values.");
218  } catch (IOException e) {
219  throw new RuntimeException(
220  "The Ant map file could not be read due to an IOException:\n" + e);
221  }
222  }
