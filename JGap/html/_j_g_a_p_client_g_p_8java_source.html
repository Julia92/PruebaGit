899  protected void evolve(IGridClientMediator a_gcmed, boolean a_receiveOnly)
900  throws Exception {
901  // Do the complete evolution cycle until end.
902  // ------------------------------------------
903  IClientFeedbackGP feedback = m_gridConfig.getClientFeedback();
904  feedback.beginWork();
905  IClientEvolveStrategyGP evolver = m_gridConfig.getClientEvolveStrategy();
906  IRequestSplitStrategyGP splitter = m_gridConfig.getRequestSplitStrategy();
907  int evolutionIndex = 0;
908  do {
909  JGAPRequestGP[] workRequests = null;
910  boolean deferRequests = false;
911  if (!a_receiveOnly) {
912  try {
913  // Care that not too much work requests are online, do a listing
914  // from time to time. If enough requests already there, don't create
915  // them any more.
916  // -----------------------------------------------------------------
917  long lastListing = m_objects.getLastListingRequestsMillis();
918  long current = System.currentTimeMillis();
919  if (current - lastListing > 60 * 60 * 1) { //60 Seconds * 60 Minutes * 1 Hour
920  // Do a listing again after 60 minutes or more.
921  // --------------------------------------------
922  MessageContext context = new MessageContext(MODULE_CS,
923  CONTEXT_WORK_REQUEST, CONTEXT_ID_EMPTY);
924  List requests = a_gcmed.listRequests(context, null, null);
925  m_objects.setLastListingRequestsMillis(current);
926  m_persister.save();
927  if (requests != null && requests.size() > 100) {
928  deferRequests = true;
929  log.info("Deferring creating and sending further requests"
930  + ", maximum reached ("
931  + requests.size() + " found).");
932  }
933  if (requests != null && requests.size() > 0) {
934  // Remove requests from database that are not in list any more.
935  // ------------------------------------------------------------
936  Map foundKeys = new HashMap();
937  Object first = requests.get(0);
938  if (String.class.isAssignableFrom(first.getClass())) {
939  // Requests of type String can be handled directly.
940  // ------------------------------------------------
941  for (Object key : requests) {
942  foundKeys.put(key, "");
943  }
944  }
945  else {
946  // Requests of type that sub classes have to handle.
947  // -------------------------------------------------
948  for (Object obj : requests) {
949  Object key = getKeyFromObject(obj);
950  if (key != null) {
951  foundKeys.put(key, "");
952  }
953  }
954  }
955  removeEntries(foundKeys, m_objects.getRequests());
956  }
957  }
958  if (!deferRequests) {
959  workRequests = sendWorkRequests(evolutionIndex, evolver, splitter,
960  feedback);
961  }
962  else {
963  // Defer creating and sending additional requests.
964  // -----------------------------------------------
965  }
966  } catch (WorkRequestsSendException wex) {
967  errorOnSendWorkRequests(wex.getCause(), wex.getWorkRequests());
968  }
969  if (!deferRequests && !afterSendWorkRequests(workRequests)) {
970  break;
971  }
972  }
973  if (this.isInterrupted()) {
974  break;
975  }
976  if (!deferRequests && !a_receiveOnly) {
977  evolver.afterWorkRequestsSent();
978  }
979  if (!m_no_comm) {
980  try {
981  receiveWorkResults(workRequests);
982  } catch (Exception ex) {
983  onErrorReceiveWorkResults(workRequests, ex);
984  }
985  }
986  if (!a_receiveOnly && !m_no_evolution) {
987  evolver.evolve();
988  // Fire listener that one evolution cycle is complete.
989  // ---------------------------------------------------
990  feedback.completeFrame(evolutionIndex);
991  evolutionIndex++;
992  // Check if evolution is finished.
993  // -------------------------------
994  if (evolver.isEvolutionFinished(evolutionIndex)) {
995  evolver.onFinished();
996  break;
997  }
998  }
999  else {
1000  a_gcmed.disconnect();
1001  log.info("Sleeping a while before beginning again...");
1002  Thread.sleep(40000);
1003  a_gcmed.connect();
1004  }
1005  } while (true);
1006  try {
1007  a_gcmed.disconnect();
1008  } catch (Exception ex) {
1009  log.error("Disconnecting from server failed!", ex);
1010  }
1011  m_gridConfig.getClientFeedback().endWork();
1012  }

1244  protected boolean resultReceived(IGPProgram a_fittest)

1247  return false;
1248  }
1252  try {
1253  Map<String, List> topAll = m_objects.getTopResults();
1254  String appid = m_gridConfig.getContext().getAppId();
1255  List<IGPProgram> topApp = topAll.get(appid);
1256  if (topApp == null) {
1257  topApp = new Vector();
1258  topAll.put(appid, topApp);
1259  }
1260  int fitter = 0;
1261  Iterator<IGPProgram> it = topApp.iterator();
1262  Object worstEntry = null;
1263  double worstFitness = -1;
1264  String norm = a_fittest.toStringNorm(0);
1265  int count = 0;
1266  double a_fitness = a_fittest.getFitnessValue();
1267  if (a_fitness > 12500) {
1268  // Store online as a backup.
1269  // -------------------------
1270  log.info("Backup up good result");
1271  String title = "fitness_" + NumberKit.niceDecimalNumber(a_fitness, 2);
1272  m_gcmed.backupResult(a_fittest, "goodResults", title);
1273  }
1274  else {
1275  if (a_fitness > 1750) {
1276  // Store not too bad result for mixing it in to new work requests.
1277  // ---------------------------------------------------------------
1278  log.info("Storing not too bad result for later reusage");
1279  String title = "ntb_fitness_"
1280  + DateKit.getNowAsString()
1281  + "_"
1282  + NumberKit.niceDecimalNumber(a_fitness, 2)
1283  + ".jgap";
1284  // Store in separate subdir.
1285  // -------------------------
1286  saveResult(m_ntbResultsDir, title, a_fittest);
1287  }
1288  } while (it.hasNext()) {
1289  IGPProgram prog = (IGPProgram) it.next();
1290  // Don't allow identical results.
1291  // ------------------------------
1292  if (prog.toStringNorm(0).equals(norm)) {
1293  fitter = 100;
1294  break;
1295  }
1296  double fitness = prog.getFitnessValue();
1297  if (Math.abs(fitness - a_fittest.getFitnessValue()) < 0.001) {
1298  fitter = 100;
1299  break;
1300  }
1301  else if (fitness >= a_fitness) {
1302  fitter++;
1303  }
1304  // Determine the worst entry for later replacement.
1305  // ------------------------------------------------
1306  if (worstEntry == null ||
1307  getConfiguration().getGPFitnessEvaluator().
1308  isFitter(worstFitness, fitness)) {
1309  worstEntry = prog;
1310  worstFitness = fitness;
1311  }
1312  count++;
1313  }
1314  boolean result = true;
1315  if (fitter < 3 || count > 3) {
1316  // Remove worst result yet and add new fit result.
1317  // -----------------------------------------------
1318  if (worstEntry != null && count >= 3) {
1320  if (!topApp.remove(worstEntry)) {
1321  log.error("Removing of worst entry failed");
1322  }
1323  }
1324  if (fitter < 3) {
1325  try {
1326  GPGenotype.checkErroneousProg(a_fittest, " add top fit", true, false);
1327  } catch (Throwable t) {
1328  log.warn("Received program not valid!");
1329  result = false;
1330  }
1331  if (result) {
1332  a_fitness = a_fittest.getFitnessValue();
1333  if (a_fitness < 1000) {
1334  result = false;
1335  }
1336  else {
1337  topApp.add(a_fittest);
1338  log.info("Added fit program, fitness: " +
1339  NumberKit.niceDecimalNumber(a_fitness, 2));

1341  result = true;
1342  }
1343  }
1344  }
1345  else {
1346  log.info(
1347  "Result not better than top results received, removed obsolete top result");
1348  result = false;
1349  }
1350  }
1351  else {
1352  log.info("Result not better than top results received");
1353  result = false;
1354  }
1355  if (result) {
1357  //m_persister.save(true, JGAPClientGP.FIELDSTOSKIP);
1358  m_persister.save();

1404  protected void presetPopulation(JGAPRequestGP a_workRequest)

1419  IGPProgram[] programs = pop.getGPPrograms();
1420 // while (index < pop.getPopSize() && pop.getGPProgram(index) != null) {
1421 // index++;
1422 // }
1423  List toAdd = new Vector();
1424  if (topApp != null && topApp.size() > 0) {
1425  // Merge top results in.
1426  // ---------------------
1427  for (IGPProgram prog : topApp) {
1428  GPGenotype.checkErroneousProg(prog, " before add preset", true);
1429  toAdd.add(prog);
1430  added++;
1431  if (added >= 3 || randGen.nextDouble() > 0.6d) {
1432  break;
1433  }
1434  }
1435  }
1436  // Merge not too bad results in.
1437  // -----------------------------
1438  String[] results = getFilenames(m_ntbResultsDir);
1439  if (results != null && results.length > 0) {
1440  int count = randGen.nextInt(Math.min(5, results.length));
1441  if (count > 0) {
1442  if (count > results.length) {
1443  count = results.length;
1444  }
1445  for (int i = 0; i < count; i++) {
1446  int index = randGen.nextInt(results.length);
1447  String filename = FileKit.addFilename(m_ntbResultsDir,
1448  results[index]);
1450  PersistableObject po = new PersistableObject(filename);
1451  IGPProgram ntb = (IGPProgram) po.load();
1452  log.info("Presetting with NTB result");
1453  added++;
1454  toAdd.add(ntb);
1455  }
1456  }
1457  }
1458  // Now merge old and new programs to one pool.
1459  // -------------------------------------------
1460  int len = programs.length;
1461  if (len > 0) {
1462  len = 0;
1463  while (len < programs.length && programs[len] != null) {
1464  len++;
1465  }
1466  IGPProgram[] programsNew = (IGPProgram[]) toAdd.toArray(new IGPProgram[] {});

1468  IGPProgram[] allPrograms = new IGPProgram[size];
1469  if (len > 0) {
1470  System.arraycopy(programs, 0, allPrograms, 0, len);
1471  }
1472  System.arraycopy(programsNew, 0, allPrograms, len, programsNew.length);
1473  pop.setGPPrograms(allPrograms);
1474  log.info("Population preset with " + added + " additional programs");
