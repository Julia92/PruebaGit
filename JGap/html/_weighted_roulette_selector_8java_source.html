106  protected synchronized void add(final IChromosome a_chromosomeToAdd) {
107  // The "roulette wheel" is represented by a Map. Each key is a
108  // Chromosome and each value is an instance of the SlotCounter inner
109  // class. The counter keeps track of the total number of slots that
110  // each chromosome is occupying on the wheel (which is equal to the
111  // combined total of their fitness values). If the Chromosome is
112  // already in the Map, then we just increment its number of slots
113  // by its fitness value. Otherwise we add it to the Map.
114  // -----------------------------------------------------------------
115  SlotCounter counter = (SlotCounter) m_wheel.get(a_chromosomeToAdd);
116  if (counter != null) {
117  // The Chromosome is already in the map.
118  // -------------------------------------
119  counter.increment();
120  }
121  else {
122  // We need to add this Chromosome and an associated SlotCounter
123  // to the map. First, we reset the Chromosome's
124  // isSelectedForNextGeneration flag to false. Later, if the
125  // Chromosome is actually selected to move on to the next
126  // generation population by the select() method, then it will
127  // be set to true.
128  // ------------------------------------------------------------
129  a_chromosomeToAdd.setIsSelectedForNextGeneration(false);
130  // We're going to need a SlotCounter. See if we can get one
131  // from the pool. If not, construct a new one.
132  // --------------------------------------------------------
133  counter = (SlotCounter) m_counterPool.acquirePooledObject();
134  if (counter == null) {
135  counter = new SlotCounter();
136  }
137  counter.reset(a_chromosomeToAdd.getFitnessValue());
138  m_wheel.put(a_chromosomeToAdd, counter);
139  }
140  }

158  public synchronized void selectChromosomes(int a_howManyToSelect,
159  Population a_to_pop) {
160  RandomGenerator generator = getConfiguration().getRandomGenerator();
161  scaleFitnessValues();
162  // Build three arrays from the key/value pairs in the wheel map: one
163  // that contains the fitness values for each chromosome, one that
164  // contains the total number of occupied slots on the wheel for each
165  // chromosome, and one that contains the chromosomes themselves. The
166  // array indices are used to associate the values of the three arrays
167  // together (eg, if a chromosome is at index 5, then its fitness value
168  // and counter values are also at index 5 of their respective arrays).
169  // -------------------------------------------------------------------
170  Set entries = m_wheel.entrySet();
171  int numberOfEntries = entries.size();
172  double[] fitnessValues = new double[numberOfEntries];
173  double[] counterValues = new double[numberOfEntries];
174  IChromosome[] chromosomes = new IChromosome[numberOfEntries];
175  m_totalNumberOfUsedSlots = 0.0d;
176  Iterator entryIterator = entries.iterator();
177  for (int i = 0; i < numberOfEntries; i++) {
178  Map.Entry chromosomeEntry = (Map.Entry) entryIterator.next();
179  IChromosome currentChromosome =
180  (IChromosome) chromosomeEntry.getKey();
181  SlotCounter currentCounter =
182  (SlotCounter) chromosomeEntry.getValue();
183  fitnessValues[i] = currentCounter.getFitnessValue();
184  counterValues[i] = fitnessValues[i] //currentCounter.getFitnessValue()
185  * currentCounter.getCounterValue();
186  chromosomes[i] = currentChromosome;
187  // We're also keeping track of the total number of slots,
188  // which is the sum of all the counter values.
189  // ------------------------------------------------------
190  m_totalNumberOfUsedSlots += counterValues[i];
191  }
192  // To select each chromosome, we just "spin" the wheel and grab
193  // whichever chromosome it lands on.
194  // ------------------------------------------------------------
195  IChromosome selectedChromosome;
196  for (int i = 0; i < a_howManyToSelect; i++) {
197  selectedChromosome = spinWheel(generator, fitnessValues, counterValues,
198  chromosomes);
199  selectedChromosome.setIsSelectedForNextGeneration(true);
200  if (a_to_pop.contains(selectedChromosome)) {
201  ICloneHandler cloner = getConfiguration().getJGAPFactory().
202  getCloneHandlerFor(selectedChromosome, null);
203  if (cloner != null) {
204  try {
205  IChromosome cloned = (IChromosome) cloner.perform(
206  selectedChromosome, null, null);
207  a_to_pop.addChromosome(cloned);
208  if (m_monitorActive) {
209  cloned.setUniqueIDTemplate(selectedChromosome.getUniqueID(), 1);
210  }
211  } catch (Exception ex) {
212  ex.printStackTrace();
213  a_to_pop.addChromosome(selectedChromosome);
214  }
215  }
216  else {
217  a_to_pop.addChromosome(selectedChromosome);
218  if (m_monitorActive) {
219  selectedChromosome.setUniqueIDTemplate(selectedChromosome.
220  getUniqueID(), 1);
221  }
222  }
223  }
224  else {
225  a_to_pop.addChromosome(selectedChromosome);
226  if (m_monitorActive) {
227  selectedChromosome.setUniqueIDTemplate("new", 1);
228  }
229  }
230  }
231  }

252  private IChromosome spinWheel(final RandomGenerator a_generator,
253  final double[] a_fitnessValues,
254  double[] a_counterValues,
255  final IChromosome[] a_chromosomes) {
256  // Randomly choose a slot on the wheel.
257  // ------------------------------------
258  double selectedSlot =
259  a_generator.nextDouble() * m_totalNumberOfUsedSlots;
260  if (selectedSlot > m_totalNumberOfUsedSlots) {
261  selectedSlot = m_totalNumberOfUsedSlots;
262  }
263  // Loop through the wheel until we find our selected slot. Here's
264  // how this works: we have three arrays, one with the fitness values
265  // of the chromosomes, one with the total number of slots on the
266  // wheel that each chromosome occupies (its counter value), and
267  // one with the chromosomes themselves. The array indices associate
268  // each of the three together (eg, if a chromosome is at index 5,
269  // then its fitness value and counter value are also at index 5 of
270  // their respective arrays).
271  //
272  // We've already chosen a random slot number on the wheel from which
273  // we want to select the Chromosome. We loop through each of the
274  // array indices and, for each one, we add the number of occupied slots
275  // (the counter value) to an ongoing total until that total
276  // reaches or exceeds the chosen slot number. When that happenes,
277  // we've found the chromosome sitting in that slot and we return it.
278  // --------------------------------------------------------------------
279  double currentSlot = 0.0d;
280  FitnessEvaluator evaluator = getConfiguration().getFitnessEvaluator();
281  boolean isFitter2_1 = evaluator.isFitter(2, 1);
282  for (int i = 0; i < a_counterValues.length; i++) {
283  // Increment our ongoing total and see if we've landed on the
284  // selected slot.
285  // ----------------------------------------------------------
286  boolean found;
287  if (isFitter2_1) {
288  // Introduced DELTA to fix bug 1449651
289  found = selectedSlot - currentSlot <= DELTA;
290  }
291  else {
292  // Introduced DELTA to fix bug 1449651
293  found = Math.abs(currentSlot - selectedSlot) <= DELTA;
294  }
295  if (found) {
296  // Remove one instance of the chromosome from the wheel by
297  // decrementing the slot counter by the fitness value resp.
298  // resetting the counter if doublette chromosomes are not
299  // allowed.
300  // -------------------------------------------------------
301  if (!getDoubletteChromosomesAllowed()) {
302  m_totalNumberOfUsedSlots -= a_counterValues[i];
303  a_counterValues[i] = 0;
304  }
305  else {
306  a_counterValues[i] -= a_fitnessValues[i];
307  m_totalNumberOfUsedSlots -= a_fitnessValues[i];
308  }
309  // Introduced DELTA to fix bug 1449651
310  if (Math.abs(m_totalNumberOfUsedSlots) < DELTA) {
311  m_totalNumberOfUsedSlots = 0.0d;
312  }
313  // Now return our selected Chromosome.
314  // -----------------------------------
315  return a_chromosomes[i];
316  }
317  else {
318  currentSlot += a_counterValues[i];
319  }
320  }
321  // We have reached here because there were rounding errors when
322  // computing with doubles or because the last entry is the right one.
323  // ------------------------------------------------------------------
324  return a_chromosomes[a_counterValues.length - 1];
325  }

344  private void scaleFitnessValues() {
345  // First, add up all the fitness values. While we're doing this,
346  // keep track of the largest fitness value we encounter.
347  // -------------------------------------------------------------
348  double largestFitnessValue = 0.0;
349  BigDecimal totalFitness = ZERO_BIG_DECIMAL;
350  Iterator counterIterator = m_wheel.values().iterator();
351  while (counterIterator.hasNext()) {
352  SlotCounter counter = (SlotCounter) counterIterator.next();
353  if (counter.getFitnessValue() > largestFitnessValue) {
354  largestFitnessValue = counter.getFitnessValue();
355  }
356  BigDecimal counterFitness = new BigDecimal(counter.getFitnessValue());
357  totalFitness = totalFitness.add(counterFitness.multiply(
358  new BigDecimal(counter.getCounterValue())));
359  }
360  // Now divide the total fitness by the largest fitness value to
361  // compute the scaling factor.
362  // ------------------------------------------------------------
363  if (largestFitnessValue > 0.000000d
364  && totalFitness.floatValue() > 0.0000001d) {
365  double scalingFactor =
366  totalFitness.divide(new BigDecimal(largestFitnessValue),
367  BigDecimal.ROUND_HALF_UP).doubleValue();
368  // Divide each of the fitness values by the scaling factor to
369  // scale them down.
370  // ----------------------------------------------------------
371  counterIterator = m_wheel.values().iterator();
372  while (counterIterator.hasNext()) {
373  SlotCounter counter = (SlotCounter) counterIterator.next();
374  counter.scaleFitnessValue(scalingFactor);
375  }
376  }
