92  public int execute_int(ProgramChromosome c, int n, Object[] args) {
93  int index;
94  // Child = index --> only relevant if not replacement for whole matrix.
95  // --------------------------------------------------------------------
96  if (m_countType != CountType.MATRIX) {
97  index = c.execute_int(n, 0, args);
98  }
99  else {
100  index = 0;
101  }
102 
103  // Retrieve the matrix.
104  // --------------------
105  char[][] matrix = getGPConfiguration().getMatrix(m_matrixName);
106  int count = 0;
107  if (matrix != null) {
108  int cols = matrix.length;
109  if(index >= cols) {
110  index = cols - 1;
111  }
112  else if (index < 0) {
113  index = 0;
114  }
115  int rows = matrix[index].length;
116  switch (m_countType) {
117  case ROW:
118  for (int col = 0; col < cols; col++) {
119  count += count(matrix[col][index]);
120  }
121  break;
122  case COLUMN:
123  for (int row = 0; row < rows; row++) {
124  count += count(matrix[index][row]);
125  }
126  break;
127  case MATRIX:
128  for (int col = 0; col < cols; col++) {
129  for (int row = 0; row < rows; row++) {
130  count += count(matrix[col][row]);
131  }
132  }
133  break;
134  case DIAGONAL:
135  if (cols != rows) {
136  throw new IllegalArgumentException("Cannot count diagonal of the"
137  +" matrix, as the matrix is not square");
138  }
139  // Index less than half the number of columns/rows: first diagonal
140  // Else: second diagonal
141  if (index < cols / 2) {
142  for (int cell = 0; cell < cols; cell++) {
143  count += count(matrix[cell][cell]);
144  }
145  } else {
146  for (int cell = cols - 1; cell > 0; cell--) {
147  count += count(matrix[cell][cell]);
148  }
149  }
150  break;
151  }
152  }
153  return count;
154  }
