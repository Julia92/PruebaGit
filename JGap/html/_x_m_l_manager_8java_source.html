169  public static Element representGenesAsElement(final Gene[] a_geneValues,
170  final Document a_xmlDocument) {
171  // Create the parent genes element.
172  // --------------------------------
173  Element genesElement = a_xmlDocument.createElement(GENES_TAG);
174  // Now add gene sub-elements for each gene in the given array.
175  // -----------------------------------------------------------
176  Element geneElement;
177  for (int i = 0; i < a_geneValues.length; i++) {
178  // Create the allele element for this gene.
179  // ----------------------------------------
180  geneElement = a_xmlDocument.createElement(GENE_TAG);
181  // Add the class attribute and set its value to the class
182  // name of the concrete class representing the current Gene.
183  // ---------------------------------------------------------
184  geneElement.setAttribute(CLASS_ATTRIBUTE,
185  a_geneValues[i].getClass().getName());
186  // Create a text node to contain the string representation of
187  // the gene's value (allele).
188  // ----------------------------------------------------------
189  Element alleleRepresentation = representAlleleAsElement(a_geneValues[i],
190  a_xmlDocument);
191  // And now add the text node to the gene element, and then
192  // add the gene element to the genes element.
193  // -------------------------------------------------------
194  geneElement.appendChild(alleleRepresentation);
195  genesElement.appendChild(geneElement);
196  }
197  return genesElement;
198  }

316  public static Gene[] getGenesFromElement(
317  Configuration a_activeConfiguration, Element a_xmlElement)
318  throws ImproperXMLException, UnsupportedRepresentationException,
319  GeneCreationException {
320  // Do some sanity checking. Make sure the XML Element isn't null and
321  // that it in fact represents a set of genes.
322  // -----------------------------------------------------------------
323  if (a_xmlElement == null ||
324  ! (a_xmlElement.getTagName().equals(GENES_TAG))) {
325  throw new ImproperXMLException(
326  "Unable to build Chromosome instance from XML Element: " +
327  "given Element is not a 'genes' element.");
328  }
329  List genes = Collections.synchronizedList(new ArrayList());
330  // Extract the nested gene elements.
331  // ---------------------------------
332  NodeList geneElements = a_xmlElement.getElementsByTagName(GENE_TAG);
333  if (geneElements == null) {
334  throw new ImproperXMLException(
335  "Unable to build Gene instances from XML Element: " +
336  "'" + GENE_TAG + "'" +
337  " sub-elements not found.");
338  }
339  // For each gene, get the class attribute so we know what class
340  // to instantiate to represent the gene instance, and then find
341  // the child text node, which is where the string representation
342  // of the allele is located, and extract the representation.
343  // -------------------------------------------------------------
344  int numberOfGeneNodes = geneElements.getLength();
345  for (int i = 0; i < numberOfGeneNodes; i++) {
346  Element thisGeneElement = (Element) geneElements.item(i);
347  thisGeneElement.normalize();
348  // Fetch the class attribute and create an instance of that
349  // class to represent the current gene.
350  // --------------------------------------------------------
351  String geneClassName =
352  thisGeneElement.getAttribute(CLASS_ATTRIBUTE);
353  Gene thisGeneObject;
354  Class geneClass = null;
355  try {
356  geneClass = Class.forName(geneClassName);
357  try {
358  Constructor constr = geneClass.getConstructor(new Class[] {
359  Configuration.class});
360  thisGeneObject = (Gene) constr.newInstance(new Object[] {
361  a_activeConfiguration});
362  } catch (NoSuchMethodException nsme) {
363  // Try it by calling method newGeneInternal.
364  // -----------------------------------------
365  Constructor constr = geneClass.getConstructor(new Class[] {});
366  thisGeneObject = (Gene) constr.newInstance(new Object[] {});
367  thisGeneObject = (Gene) PrivateAccessor.invoke(thisGeneObject,
368  "newGeneInternal", new Class[] {}, new Object[] {});
369  }
370  } catch (Throwable e) {
371  throw new GeneCreationException(geneClass, e);
372  }
373  // Find the text node and fetch the string representation of
374  // the allele.
375  // ---------------------------------------------------------
376  NodeList children = thisGeneElement.getChildNodes();
377  int childrenSize = children.getLength();
378  String alleleRepresentation = null;
379  for (int j = 0; j < childrenSize; j++) {
380  Element alleleElem = (Element) children.item(j);
381  if (alleleElem.getTagName().equals(ALLELE_TAG)) {
382  alleleRepresentation = alleleElem.getAttribute("value");
383  }
384  if (children.item(j).getNodeType() == Node.TEXT_NODE) {
385  // We found the text node. Extract the representation.
386  // ---------------------------------------------------
387  alleleRepresentation = children.item(j).getNodeValue();
388  break;
389  }
390  }
391  // Sanity check: Make sure the representation isn't null.
392  // ------------------------------------------------------
393  if (alleleRepresentation == null) {
394  throw new ImproperXMLException(
395  "Unable to build Gene instance from XML Element: " +
396  "value (allele) is missing representation.");
397  }
398  // Now set the value of the gene to that reflect the
399  // string representation.
400  // -------------------------------------------------
401  try {
402  thisGeneObject.setValueFromPersistentRepresentation(
403  alleleRepresentation);
404  } catch (UnsupportedOperationException e) {
405  throw new GeneCreationException(
406  "Unable to build Gene because it does not support the " +
407  "setValueFromPersistentRepresentation() method.");
408  }
409  // Finally, add the current gene object to the list of genes.
410  // ----------------------------------------------------------
411  genes.add(thisGeneObject);
412  }
413  return (Gene[]) genes.toArray(new Gene[genes.size()]);
414  }
