150  private double computeFitness(IGPProgram a_program, Variable vx) {
151  double error = 0.0f;
152  Object[] noargs = new Object[0];
153  // Initialize local stores.
154  // ------------------------
155  a_program.getGPConfiguration().clearStack();
156  a_program.getGPConfiguration().clearMemory();
157  // Compute fitness for each program.
158  // ---------------------------------
159  for (int i = 2; i < 15; i++) {
160  for (int j = 0; j < a_program.size(); j++) {
161  vx.set(new Integer(i));
162  try {
163  try {
164  // Only evaluate after whole GP program was run.
165  // ---------------------------------------------
166  if (j == a_program.size() - 1) {
167  double result = a_program.execute_int(j, noargs);
168  error += Math.abs(result - fib_iter(i));
169  }
170  else {
171  a_program.execute_void(j, noargs);
172  }
173  } catch (IllegalStateException iex) {
174  error = Double.MAX_VALUE / 2;
175  break;
176  }
177  } catch (ArithmeticException ex) {
178  System.out.println("Arithmetic Exception with x = " + i);
179  System.out.println(a_program.getChromosome(j));
180  throw ex;
181  }
182  }
183  }
184  return error;
185  }

187  private int fib_iter(int a_index) {
188  // 1
189  if (a_index == 0 || a_index == 1) {
190  return 1;
191  }
192  // 2
193  int a = 1; //Store("mem0", Constant(1))
194  int b = 1; //Store("mem1", Constant(1))
195  int x = 0; //Store("mem2", Constant(0))
196  // 3
197  for (int i = 2; i <= a_index; i++) { //FORX (Subprogram(A;B;C))
198  x = a + b; // A: AddAndStore(Read("mem0"),Read("mem1"),"mem2")
199  a = b; //B: TransferMemory("mem1","mem0")
200  b = x; //C: TransferMemory("mem2","mem1")
201  }
202  return x; //Read("mem2")
203  }
