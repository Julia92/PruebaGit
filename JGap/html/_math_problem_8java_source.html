99  new Multiply(conf, CommandGene.FloatClass),

105  new Pow(conf, CommandGene.FloatClass),

107  // ADF-relevant:
108  // Construct a reference to the ADF defined in the second nodeset
109  // which has index 1 (second parameter of ADF-constructor).
110  // Furthermore, the ADF expects three input parameters (see argTypes[1])
111  new ADF(conf, 1 , 3),
112  },
113  // ADF-relevant:
114  // and now the definition of ADF(1)
115  {
116  new Add3(conf, CommandGene.FloatClass),
117  }
118  };

120  // function/formula to evolve by the GP.
121  // -----------------------------------------------------------------------

124  // ---------------------------------------------------------------

203  public double computeRawFitness(final IGPProgram ind) {
204  double error = 0.0f;
205  Object[] noargs = new Object[0];
206  // Evaluate function for input numbers 0 to 20.
207  // --------------------------------------------
208  for (int i = 0; i < 20; i++) {
209  // Provide the variable X with the input number.
210  // See method create(), declaration of "nodeSets" for where X is
211  // defined.
212  // -------------------------------------------------------------
213  vx.set(x[i]);
214  try {
215  // Execute the GP program representing the function to be evolved.
216  // As in method create(), the return type is declared as float (see
217  // declaration of array "types").
218  // ----------------------------------------------------------------
219  double result = ind.execute_float(0, noargs);
220  // Sum up the error between actual and expected result to get a defect
221  // rate.
222  // -------------------------------------------------------------------
223  error += Math.abs(result - y[i]);
224  // If the error is too high, stop evlauation and return worst error
225  // possible.
226  // ----------------------------------------------------------------
227  if (Double.isInfinite(error)) {
228  return Double.MAX_VALUE;
229  }
230  } catch (ArithmeticException ex) {
231  // This should not happen, some illegal operation was executed.
232  // ------------------------------------------------------------
233  System.out.println("x = " + x[i].floatValue());
234  System.out.println(ind);
235  throw ex;
236  }
237  }
238  // In case the error is small enough, consider it perfect.
239  // -------------------------------------------------------
240  if (error < 0.001) {
241  error = 0.0d;
242  }
243  return error;
244  }
245  }
