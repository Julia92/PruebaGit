64  // Start with a DefaultConfiguration, which comes setup with the
65  // most common settings.
66  // -------------------------------------------------------------
67  Configuration conf = new DefaultConfiguration();
68  // Care that the fittest individual of the current population is
69  // always taken to the next generation.
70  // Consider: With that, the pop. size may exceed its original
71  // size by one sometimes!
72  // -------------------------------------------------------------
73  conf.setPreservFittestIndividual(true);
74  conf.setKeepPopulationSizeConstant(false);
75  // Set the fitness function we want to use, which is our
76  // MinimizingMakeChangeFitnessFunction. We construct it with
77  // the target amount of change passed in to this method.
78  // ---------------------------------------------------------
79  FitnessFunction myFunc =
80  new SampleFitnessFunction(a_targetChangeAmount);
81  conf.setFitnessFunction(myFunc);
82  if (a_doMonitor) {
83  // Turn on monitoring/auditing of evolution progress.
84  // --------------------------------------------------
85  m_monitor = new EvolutionMonitor();
86  conf.setMonitor(m_monitor);
87  }
88  // Now we need to tell the Configuration object how we want our
89  // Chromosomes to be setup. We do that by actually creating a
90  // sample Chromosome and then setting it on the Configuration
91  // object. As mentioned earlier, we want our Chromosomes to each
92  // have four genes, one for each of the coin types. We want the
93  // values (alleles) of those genes to be integers, which represent
94  // how many coins of that type we have. We therefore use the
95  // IntegerGene class to represent each of the genes. That class
96  // also lets us specify a lower and upper bound, which we set
97  // to sensible values for each coin type.
98  // --------------------------------------------------------------
99  Gene[] sampleGenes = new Gene[4];
100  sampleGenes[0] = new IntegerGene(conf, 0, 98); // Wasser
101  sampleGenes[1] = new IntegerGene(conf, 0, 98); // Zucker
102  sampleGenes[2] = new IntegerGene(conf, 0, 98); // Saft1
103  sampleGenes[3] = new IntegerGene(conf, 0, 98); // Saft2
104  IChromosome sampleChromosome = new Chromosome(conf, sampleGenes);
105  conf.setSampleChromosome(sampleChromosome);
106  // Finally, we need to tell the Configuration object how many
107  // Chromosomes we want in our population. The more Chromosomes,
108  // the larger number of potential solutions (which is good for
109  // finding the answer), but the longer it will take to evolve
110  // the population (which could be seen as bad).
111  // ------------------------------------------------------------
112  conf.setPopulationSize(80);
113 
114  // Now we initialize the population randomly, anyway (as an example only)!
115  // If you want to load previous results from file, remove the next line!
116  // -----------------------------------------------------------------------
117  Genotype population = Genotype.randomInitialGenotype(conf);
118  // Evolve the population. Since we don't know what the best answer
119  // is going to be, we just evolve the max number of times.
120  // ---------------------------------------------------------------
121  long startTime = System.currentTimeMillis();
122  for (int i = 0; i < MAX_ALLOWED_EVOLUTIONS; i++) {
123  if (!uniqueChromosomes(population.getPopulation())) {

125  }
126  if(m_monitor != null) {
127  population.evolve(m_monitor);
128  }
129  else {
130  population.evolve();
131  }
132  }

137  // resume where it stopped before! --> this is completely optional.
138  // ---------------------------------------------------------------------
139 
140  // Display the best solution we found.
141  // -----------------------------------
142  IChromosome bestSolutionSoFar = population.getFittestChromosome();
