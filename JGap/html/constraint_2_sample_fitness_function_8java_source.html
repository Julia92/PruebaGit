50  public double evaluate(IChromosome a_subject) {
51  // Take care of the fitness evaluator. It could either be weighting higher
52  // fitness values higher (e.g.DefaultFitnessEvaluator). Or it could weight
53  // lower fitness values higher, because the fitness value is seen as a
54  // defect rate (e.g. DeltaFitnessEvaluator)
55  boolean defaultComparation = a_subject.getConfiguration().
56  getFitnessEvaluator().isFitter(1, 2);
57  double changeAmount = amountOfChange(a_subject);
58  int totalCoins = getTotalNumberOfCoins(a_subject);
59  int changeDifference = Math.abs(m_targetAmount - (int)changeAmount);
60  double fitness;
61  if (defaultComparation) {
62  fitness = 0.0d;
63  }
64  else {
65  fitness = MAX_BOUND / 2;
66  }
67  // Step 1: Determine distance of amount represented by solution from
68  // the target amount. If the change difference is greater than zero we
69  // will divide one by the difference in change between the
70  // solution amount and the target amount. That will give the desired
71  // effect of returning higher values for amounts closer to the target
72  // amount and lower values for amounts further away from the target
73  // amount.
74  // In the case where the change difference is zero it means that we have
75  // the correct amount and we assign a higher fitness value.
76  // ---------------------------------------------------------------------
77  if (defaultComparation) {
78  fitness += changeDifferenceBonus(MAX_BOUND / 2, changeDifference);
79  }
80  else {
81  fitness -= changeDifferenceBonus(MAX_BOUND / 2, changeDifference);
82  }
83  // Step 2: We divide the fitness value by a penalty based on the number of
84  // coins. The higher the number of coins the higher the penalty and the
85  // smaller the fitness value.
86  // And inversely the smaller number of coins in the solution the higher
87  // the resulting fitness value.
88  // -----------------------------------------------------------------------
89  if (defaultComparation) {
90  fitness -= computeCoinNumberPenalty(MAX_BOUND / 2, totalCoins);
91  }
92  else {
93  fitness += computeCoinNumberPenalty(MAX_BOUND / 2, totalCoins);
94  }
95  // Make sure fitness value is always positive.
96  // -------------------------------------------
97  return Math.max(1.0d, fitness);
98  }

109  protected double changeDifferenceBonus(double a_maxFitness,
110  int a_changeDifference) {
111  if (a_changeDifference == 0) {
112  return a_maxFitness;
113  }
114  else {
115  // we arbitrarily work with half of the maximum fitness as basis for non-
116  // optimal solutions (concerning change difference)
117  if (a_changeDifference * a_changeDifference >= a_maxFitness / 2) {
118  return 0.0d;
119  }
120  else {
121  return a_maxFitness / 2 - a_changeDifference * a_changeDifference;
122  }
123  }

137  protected double computeCoinNumberPenalty(double a_maxFitness, int a_coins) {
138  if (a_coins == 1) {
139  // we know the solution cannot have less than one coin
140  return 0;
141  }
142  else {
143  // The more coins the more penalty, but not more than the maximum fitness
144  // value possible. Let's avoid linear behavior and use
145  // exponential penalty calculation instead
146  return (Math.min(a_maxFitness, a_coins * a_coins));
147  }
