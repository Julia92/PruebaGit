557  public void calcFitness() {
558  double totalFitness = 0.0d;
559  GPPopulation pop = getGPPopulation();
560  IGPProgram best = null;
561  IGPFitnessEvaluator evaluator = getGPConfiguration().getGPFitnessEvaluator();
562  m_bestFitness = FitnessFunction.NO_FITNESS_VALUE;
563  boolean bestPreserved = false;
564  for (int i = 0; i < pop.size() && pop.getGPProgram(i) != null; i++) {
565  IGPProgram program = pop.getGPProgram(i);
570  double fitness;
571  try {
572  fitness = program.getFitnessValue();
573  } catch (IllegalStateException iex) {
574  fitness = Double.NaN;
575  }
576  // Don't acceppt Infinity or NaN as a result.
577  // ------------------------------------------
578  if (Double.isInfinite(fitness) || Double.isNaN(fitness)) {
579  continue;
580  }
581  if (best == null || evaluator.isFitter(fitness, m_bestFitness)) {
582  best = program;
583  m_bestFitness = fitness;
584  if (m_allTimeBest != null && !bestPreserved) {
585  if (best.toStringNorm(0).equals(m_allTimeBest.toStringNorm(0))) {
586  bestPreserved = true;
587  }
588  }
589  }
590  totalFitness += fitness;
591  }
592  m_totalFitness = totalFitness;
593  best = pop.determineFittestProgram();
594  if (best != null) {
595  m_bestFitness = best.getFitnessValue();
597  if (m_allTimeBest == null
598  || evaluator.isFitter(m_bestFitness, m_allTimeBestFitness)) {
599  pop.setChanged(true);
600  try {
601  ICloneHandler cloner = getGPConfiguration().getJGAPFactory().
602  getCloneHandlerFor(best, null);
603  if (cloner == null) {
604  m_allTimeBest = best;
605  if (!m_cloneWarningGPProgramShown) {
606  LOGGER.info("Warning: cannot clone instance of "
607  + best.getClass());
608  m_cloneWarningGPProgramShown = true;
609  }
610  }
611  else {
612  m_allTimeBest = (IGPProgram) cloner.perform(best, null, null);
613  }
614  } catch (Exception ex) {
615  m_allTimeBest = best;
616  ex.printStackTrace();
617  }
618  m_allTimeBestFitness = m_bestFitness;
619  // Fire an event to indicate a new best solution.
620  // ----------------------------------------------
622  try {
623  getGPConfiguration().getEventManager().fireGeneticEvent(
624  new GeneticEvent(GeneticEvent.GPGENOTYPE_NEW_BEST_SOLUTION, this));
625  } catch (IllegalArgumentException iex) {
626  iex.printStackTrace();
628  }
629  if (m_verbose) {
630  // Output the new best solution found.
631  // -----------------------------------
632  outputSolution(m_allTimeBest);
633  }
634  }
635  }
636  if (!bestPreserved && m_allTimeBest != null) {
637  addFittestProgram(m_allTimeBest);
638  }

696  public void evolve() {
697  try {
698  int popSize = getGPConfiguration().getPopulationSize();
699  GPPopulation oldPop = getGPPopulation();
700  GPPopulation newPopulation = new GPPopulation(oldPop, false);
701  if (m_fittestToAdd != null) {
702  newPopulation.addFittestProgram(m_fittestToAdd);
703  m_fittestToAdd = null;
704  }
705  float val;
706  RandomGenerator random = getGPConfiguration().getRandomGenerator();
707  GPConfiguration conf = getGPConfiguration();
708  // Determine how many new individuals will be added to the new generation.
709  // -----------------------------------------------------------------------
710  int popSize1 = (int) Math.round(popSize * (1 - conf.getNewChromsPercent()));
711  double crossProb = conf.getCrossoverProb()
712  / (conf.getCrossoverProb() + conf.getReproductionProb());
713  int crossover = 0;
714  int reproduction = 0;
715  int creation = 0;
716  checkErroneousPop(getGPPopulation(), " (before evolution)", true);
717  final int maxTries = getGPConfiguration().getProgramCreationMaxtries();
718  // Do crossing over.
719  // -----------------
720  for (int i = 0; i < popSize1; i++) {
721  // Clear the stack for each GP program.
722  // ------------------------------------
723  getGPConfiguration().clearStack();
724  val = random.nextFloat();
725  // Note that if we only have one slot left to fill, we don't do
726  // crossover, but fall through to reproduction.
727  // ------------------------------------------------------------
728  if (i < popSize - 1 && val < crossProb) {
729  crossover++;
730  // Actually do the crossover here.
731  // -------------------------------
732  IGPProgram i1 = conf.getSelectionMethod().select(this);
733  IGPProgram i2 = conf.getSelectionMethod().select(this);
734  int tries = 0;
735  do {
736  try {
737  checkErroneousProg(i1,
738  " at start of evolution (index " + i +
739  "/01)", false);
740  if (i1 != i2) {
741  // Crossing over the a program with itself does not lead anywhere.
742  // ---------------------------------------------------------------
743  checkErroneousProg(i2,
744  " at start of evolution (index " + i +
745  "/02)", false);
746  IGPProgram[] newIndividuals = conf.getCrossMethod().operate(
747  i1,
748  i2);
749  newPopulation.setGPProgram(i, newIndividuals[0]);
750  newPopulation.setGPProgram(i + 1, newIndividuals[1]);
751  try {
752  checkErroneousProg(newIndividuals[0],
753  " at start of evolution (index " + i +
754  "/11)", false);
755  } catch (RuntimeException t) {
756  writeToFile(i1, i2, newIndividuals[0],
757  "Error in first X-over program");
758  throw t;
759  }
760  try {
761  checkErroneousProg(newIndividuals[1],
762  " at start of evolution (index " + i +
763  "/12)", false);
764  } catch (RuntimeException t) {
765  writeToFile(i1, i2, newIndividuals[1],
766  "Error in second X-over program");
767  throw t;
768  }
769  }
770  else {
771  newPopulation.setGPProgram(i, i1);
772  newPopulation.setGPProgram(i + 1, i2);
773  }
774  i++;
775  break;
776  } catch (IllegalStateException iex) {
777  tries++;
778  if ( (maxTries > 0 && tries >= maxTries) || tries > 40) {
779  if (!getGPConfiguration().isMaxNodeWarningPrinted()) {
780  LOGGER.error(
781  "Warning: Maximum number of nodes allowed may be too small");
782  getGPConfiguration().flagMaxNodeWarningPrinted();
783  }
784  // Try cloning a previously generated valid program.
785  // -------------------------------------------------
786  IGPProgram program = cloneProgram(getGPConfiguration().
787  getPrototypeProgram());
788  if (program != null) {
789  newPopulation.setGPProgram(i++, program);
790  program = cloneProgram(getGPConfiguration().
791  getPrototypeProgram());
792  newPopulation.setGPProgram(i, program);
793  break;
794  }
795  else {
796  throw new IllegalStateException(iex.getMessage());
797  }
798  }
799  }
800  } while (true)
801  ;
802  }
803  else { //if (val < conf.getCrossoverProb() + conf.getReproductionProb()) {
804  // Reproduction only.
805  // ------------------
806  reproduction++;
807  newPopulation.setGPProgram(i, conf.getSelectionMethod().select(this));
808  }
809  }
810  // Add new random programs.
811  // ------------------------
812  for (int i = popSize1; i < popSize; i++) {
813  creation++;
814  // Randomly determine depth between minInitDepth and maxInitDepth.
815  // ---------------------------------------------------------------
816  int depth = conf.getMinInitDepth()
817  + random.nextInt(conf.getMaxInitDepth() - conf.getMinInitDepth()
818  + 1);
819  int tries = 0;
820  int nogc = 0;
821  do {
822  try {
823  // Randomize grow option as growing produces a valid program
824  // more likely than the full mode.
825  // ---------------------------------------------------------
826  boolean grow;
827  if (i % 2 == 0 || random.nextInt(8) > 6) {
828  grow = true;
829  }
830  else {
831  grow = false;
832  }
836  IGPProgram program = newPopulation.create(i, m_types, m_argTypes,
837  m_nodeSets, m_minDepths, m_maxDepths, depth, grow,
838  m_maxNodes, m_fullModeAllowed, tries);
839  newPopulation.setGPProgram(i, program);
840  checkErroneousProg(program,
841  " when adding a program, evolution (index " + i +
842  ")", true);
843  LOGGER.debug("Added new GP program (depth parameter: "
844  + depth
845  + ", "
846  + tries
847  + " tries)");
848  break;
849  } catch (IllegalStateException iex) {
850  tries++;
851  nogc++;
856  if ( (maxTries > 0 && tries > maxTries) || tries > 40) {
857  LOGGER.debug(
858  "Creating random GP program failed (depth "
859  + depth
860  + ", "

863  // Try cloning a previously generated valid program.
864  // -------------------------------------------------
865  IGPProgram program = cloneProgram(getGPConfiguration().
866  getPrototypeProgram());

868  // Cloning worked.
869  // ---------------

892  }

895  + reproduction + " reproductions, "
896  + creation + " creations");

1090  * the chromosomes within the population. The logic for the hashcode is
1091  *
1092  * Step Result
1093  * ---- ------
1094  * 1 31*0 + hashcode_0 = y(1)
1095  * 2 31*y(1) + hashcode_1 = y(2)
1096  * 3 31*y(2) + hashcode_2 = y(3)
1097  * n 31*y(n-1) + hashcode_n-1 = y(n)
1098  *
1099  * Each hashcode is a number and the binary equivalent is computed and
1100  * returned.
1101  * @return the computed hashcode
1102  *
1103  * @author Klaus Meffert
1104  * @since 3.0

1109  int twopower = 1;
1110  // For empty genotype we want a special value different from other hashcode
1111  // implementations.

1114  for (i = 0; i < size; i++, twopower = 2 * twopower) {
1115  prog = getGPPopulation().getGPProgram(i);
1116  localHashCode = 31 * localHashCode + prog.hashCode();
