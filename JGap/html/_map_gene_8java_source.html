264  // ----------------------------------------------------
265  if (valueRepresentation.equals("null")) {

267  }
268  else {

280  // ---------------
281  String s = tokenizer.nextToken();

283  int lastWasOpening = 0;

286  String valueClass = null;

288  String element = tokenizer.nextToken(",");
