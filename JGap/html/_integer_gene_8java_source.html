294  if (getAllele() != null) {
295  Integer i_value = ( (Integer) getAllele());
296  // If the value exceeds either the upper or lower bounds, then
297  // map the value to within the legal range. To do this, we basically
298  // calculate the distance between the value and the integer min,
299  // determine how many bounds units that represents, and then add
300  // that number of units to the upper bound.
301  // -----------------------------------------------------------------
302  if (i_value.intValue() > m_upperBounds ||
303  i_value.intValue() < m_lowerBounds) {
304  RandomGenerator rn;
