157  private int[][] readTrail(String a_filename)
158  throws Exception {
159  LineNumberReader lnr;
160  try {
161  lnr = new LineNumberReader(new FileReader(a_filename));
162  } catch (FileNotFoundException fex) {
163  throw new FileNotFoundException("File not found: " +
164  new File(".").getAbsolutePath() +
165  a_filename);
166  }
167  // Read dimensions of trail.
168  // -------------------------
169  try {
170  StringTokenizer st = new StringTokenizer(lnr.readLine());
171  m_maxx = Integer.parseInt(st.nextToken());
172  m_maxy = Integer.parseInt(st.nextToken());
173  int[][] result = new int[m_maxx][m_maxy];
174  int y;
175  foodAvail = 0;
176  for (y = 0; y < m_maxy; y++) {
177  String s = lnr.readLine();
178  if (s == null) {
179  throw new RuntimeException("Ant trail file ended prematurely");
180  }
181  int x;
182  for (x = 0; x < s.length(); x++) {
183  if (s.charAt(x) == ' ') {
184  result[x][y] = AntMap.EMPTY;
185  }
186  else if (s.charAt(x) == '#') {
187  result[x][y] = AntMap.FOOD;
188  foodAvail++;
189  }
190  else if (s.charAt(x) == '.') {
191  result[x][y] = AntMap.TRAIL;
192  }
193  else {
194  throw new RuntimeException("Bad character '" + s.charAt(x) +
195  "' on line number " + lnr.getLineNumber() +
196  " of the Ant trail file.");
197  }
198  }
199  // fill out rest of X's
200  for (int z = x; z < m_maxx; z++) {
201  result[z][y] = AntMap.EMPTY;
202  }
203  }
204  // fill out rest of Y's
205  for (int z = y; z < m_maxy; z++) {
206  for (int x = 0; x < m_maxx; x++) {
207  result[x][z] = AntMap.EMPTY;
208  }
209  }
210  return result;
211  } catch (NumberFormatException e) {
212  throw new RuntimeException(
213  "The Ant trail file does not begin with x and y integer values.");
214  } catch (IOException e) {
215  throw new RuntimeException(
216  "The Ant trail file could not be read due to an IOException:\n" + e);
217  }
218  }

311  public double computeRawFitness(final IGPProgram a_program) {
312  double error = 0.0f;
313  Object[] noargs = new Object[0];
314  // Initialize local stores.
315  // ------------------------
316  a_program.getGPConfiguration().clearStack();
317  a_program.getGPConfiguration().clearMemory();
318  AntMap antmap = new AntMap(m_map, m_maxMoves);
319  a_program.setApplicationData(antmap);
320  try {
321  // Execute the program.
322  // --------------------
323  a_program.execute_void(0, noargs);
324  // Determine success of individual.
325  // --------------------------------
326  antmap = (AntMap) a_program.getApplicationData();
327  // The remaining food is the defect rate here.
328  // -------------------------------------------
329  int foodTaken = antmap.getFoodTaken(); // countFood(antmap);
330  error = (VALUE1 + totalFood - foodTaken) * 4;
331  if (a_program.getGPConfiguration().stackSize() > 0) {
332  error = GPFitnessFunction.MAX_FITNESS_VALUE;
333  }
334  if (error < 0.000001) {
335  error = 0.0d;
336  }
337  else if (error < GPFitnessFunction.MAX_FITNESS_VALUE) {
338  // Add penalty for longer trails.
339  // ------------------------------
340  int moves = antmap.getMoveCount();
341  error = error + moves * 1.5;
342  }
343  } catch (IllegalStateException iex) {
344  error = GPFitnessFunction.MAX_FITNESS_VALUE;
345  }
346  return error;
347  }
348  }
