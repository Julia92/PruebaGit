222  public void applyMutation(final int a_index, final double a_percentage) {

225  // ----------------------------------------------------------------------

229  if (m_immutable[a_index].contains(this)) {

231  }

233  }

235  // Following commented out because if only very few valid states exist, it

244  // --------------------------------------------------------------------
