100  public void execute_void(ProgramChromosome c, int n, Object[] args) {
101  check(c);
102  int evaluation = 0;
103  int index = 0;
104  for (int x = 0; x < Board.WIDTH; x++) {
105  for (int y = 0; y < Board.HEIGHT; y++) {
106  // Add 1 to board value to eliminate zeros.
107  // ----------------------------------------
108  int boardValue = m_board.readField(x + 1, y + 1) + 1;
109  // Adapt base of exponentiation to get unique values.
110  // --------------------------------------------------
111  evaluation += Math.pow(3 + (index * 2), (boardValue));
112  index++;
113  }
114  }

120  else {
121  memoryIndex = m_index;
122  }

124  }
