268  protected IHandler findHandlerFor(final Object a_obj,
269  final Class a_class,
270  final List a_list,
271  final IHandler a_default,
272  final String a_listID) {
273  String key = null;
274  String key1, key2;
275  if (m_useCaching) {
276  // Construct key for cache lookup:
277  // Class name of list + a_class-Name + a_obj.hashCode()
278  // ----------------------------------------------------
279  if (a_class == null) {
280  key1 = "null";
281  }
282  else {
283  key1 = a_class.getName();
284  }
285  if (a_obj == null) {
286  key2 = "null";
287  }
288  else {
289  key2 = a_obj.getClass().getName();
290  }
291  key = a_listID + "/" + key1 + "/" + key2;
292  // Lookup cache.
293  // -------------
294  Object handler = m_cache.get(key);
295  if (handler != null) {
296  return (IHandler) handler;
297  }
298  // Not found in cache. Search initially.
299  // -------------------------------------
300  }
301  IHandler result = null;
302  Iterator it = a_list.iterator();
303  while (it.hasNext()) {
304  IHandler initer = (IHandler) it.next();
305  if (initer.isHandlerFor(a_obj, a_class)) {
306  result = initer;
307  break;
308  }
309  }
310  if (result == null) {
311  // No registered handler found. Try the default handler.
312  // -----------------------------------------------------
313  if (a_default != null) {
314  if (a_default.isHandlerFor(a_obj, a_class)) {
315  result = a_default;
316  }
317  }
318  }
319  if (m_useCaching) {
320  // Add to cache.
321  // -------------
322  if (result != null) {
323  m_cache.put(key, result);
324  }
325  }
326  return result;
327  }
