194  public static void updateModuleLibrary(String BASE_URL, String a_moduleName,
195  String a_libDir, String a_workDir)
196  throws Exception {
197  String currentVersion;
198  String JGAPVersionNeeded;
199  String filename;
200  boolean isCoreModule;
201  if (a_moduleName.equalsIgnoreCase("evolutionDistributed")) {
203  filename = "evdistr.jar";
204  isCoreModule = false;
205  }
206  else if (a_moduleName.equalsIgnoreCase("jgap")) {
208  filename = "jgap.jar";
209  isCoreModule = true;
210  }
211  else if (a_moduleName.equalsIgnoreCase("rjgrid")) {
213  filename = "rjgrid.zip";
214  isCoreModule = true;
215  }
216  else {
217  throw new IllegalArgumentException("Unknown module " + a_moduleName);
218  }
219  filename = a_libDir + filename;
220  // Determine if file available locally
221  if (!FileKit.existsFile(filename)) {
222  currentVersion = "none";
223  System.out.println(" File not found: " + filename);
224  }
225  else {
226  // If file exists: determine version of module
227  if (isCoreModule) {
228  currentVersion = FileKit.getVersionOfJGAP(filename);
229  }
230  else {
231  currentVersion = FileKit.getVersionOfModule(filename);
232  // Also determine minimal version of JGAP library that is
233  // required for the module to work.
234  // -------------------------------------------------------
235  JGAPVersionNeeded = FileKit.getVersionOfJGAP(filename);
236  }
237  }
238  // ask server if update required
239  VersionInfo versionInfo = GridKit.isUpdateAvailable(BASE_URL, a_moduleName,
240  currentVersion);
241  if (versionInfo.currentVersion == null) {
242  System.out.println("Module " + a_moduleName + " is unknown");
243  return;
244  }
245  if (!currentVersion.equals(versionInfo.currentVersion)) {
246  System.out.println("Newer module " + a_moduleName + " with version "
247  + versionInfo.currentVersion
248  + " available");
249  String workDir = a_workDir;
250  // Download the newest module version
251  filename = GridKit.retrieveModule(BASE_URL, versionInfo, workDir);
252  if (filename != null) {
253  // Do the update
254  GridKit.updateModule(filename, workDir, a_libDir);
255  }
256  }
257  else {
258  System.out.println("Module " + a_moduleName + " with version " +
259  currentVersion + " is up-to-date");
260  }
261  }
