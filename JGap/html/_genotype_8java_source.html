397  public boolean equals(Object a_other) {
398  try {
399  // First, if the other Genotype is null, then they're not equal.
400  // -------------------------------------------------------------
401  if (a_other == null) {
402  return false;
403  }
404  Genotype otherGenotype = (Genotype) a_other;
405  // First, make sure the other Genotype has the same number of
406  // chromosomes as this one.
407  // ----------------------------------------------------------
408  if (getPopulation().size() != otherGenotype.getPopulation().size()) {
409  return false;
410  }
411  // Next, prepare to compare the chromosomes of the other Genotype
412  // against the chromosomes of this Genotype. To make this a lot
413  // simpler, we first sort the chromosomes in both this Genotype
414  // and the one we're comparing against. This won't affect the
415  // genetic algorithm (it doesn't care about the order), but makes
416  // it much easier to perform the comparison here.
417  // --------------------------------------------------------------
418  Collections.sort(getPopulation().getChromosomes());
419  Collections.sort(otherGenotype.getPopulation().getChromosomes());
420  for (int i = 0; i < getPopulation().size(); i++) {
421  if (! (getPopulation().getChromosome(i).equals(
422  otherGenotype.getPopulation().getChromosome(i)))) {
423  return false;
424  }
425  }
426  return true;
427  } catch (ClassCastException e) {
428  return false;

442  protected void applyNaturalSelectors(
443  boolean a_processBeforeGeneticOperators) {
445  try {
446  // Process all natural selectors applicable before executing the
447  // genetic operators (reproduction, crossing over, mutation...).
448  // -------------------------------------------------------------
449  int selectorSize = m_activeConfiguration.getNaturalSelectorsSize(
450  a_processBeforeGeneticOperators);
451  if (selectorSize > 0) {
452  int population_size = m_activeConfiguration.getPopulationSize();
453  if (a_processBeforeGeneticOperators) {
454  // Only select part of the previous population into this generation.
455  // -----------------------------------------------------------------
456  population_size = (int) Math.round(population_size *
457  getConfiguration().getSelectFromPrevGen());
458  }
459  int single_selection_size;
460  Population new_population;
461  new_population = new Population(m_activeConfiguration,
462  population_size);
463  NaturalSelector selector;
464  // Repopulate the population of chromosomes with those selected
465  // by the natural selector. Iterate over all natural selectors.
466  // ------------------------------------------------------------
467  for (int i = 0; i < selectorSize; i++) {
468  selector = m_activeConfiguration.getNaturalSelector(
469  a_processBeforeGeneticOperators, i);
470  if (i == selectorSize - 1 && i > 0) {
471  // Ensure the last NaturalSelector adds the remaining Chromosomes.
472  // ---------------------------------------------------------------
473  single_selection_size = population_size - getPopulation().size();
474  }
475  else {
476  single_selection_size = population_size / selectorSize;
477  }
478  // Do selection of Chromosomes.
479  // ----------------------------
480  selector.select(single_selection_size, getPopulation(),
481  new_population);
482  // Clean up the natural selector.
483  // ------------------------------
484  selector.empty();
485  }
486  setPopulation(new Population(m_activeConfiguration));
487  getPopulation().addChromosomes(new_population);
488  }
489  } catch (InvalidConfigurationException iex) {
490  // This exception should never be reached
491  throw new IllegalStateException(iex.getMessage());
492  }
493  }

539  /***
540  * Hashcode function for the genotype, tries to create a unique hashcode for
541  * the chromosomes within the population. The logic for the hashcode is
542  *
543  * Step Result
544  * ---- ------
545  * 1 31*0 + hashcode_0 = y(1)
546  * 2 31*y(1) + hashcode_1 = y(2)
547  * 3 31*y(2) + hashcode_2 = y(3)
548  * n 31*y(n-1) + hashcode_n-1 = y(n)
549  *
550  * @return the computed hashcode
551  *
552  * @author vamsi
553  * @since 2.1
554  */
555  public int hashCode() {
556  int i, size = getPopulation().size();
557  IChromosome s;
558  int twopower = 1;
559  // For empty genotype we want a special value different from other hashcode
560  // implementations.
561  // ------------------------------------------------------------------------
562  int localHashCode = -573;
563  for (i = 0; i < size; i++, twopower = 2 * twopower) {
564  s = getPopulation().getChromosome(i);
565  localHashCode = 31 * localHashCode + s.hashCode();
566  }
567  return localHashCode;
568  }
