450  public int compareTo(Object a_other) {
451  // First, if the other gene (or its value) is null, then this is
452  // the greater allele. Otherwise, just use the contained genes' compareTo
453  // method to perform the comparison.
454  // ---------------------------------------------------------------
455  if (a_other == null) {
456  return 1;
457  }
458  if (! (a_other instanceof CompositeGene)) {
459  return this.getClass().getName().compareTo(a_other.getClass().getName());
460  }
461  CompositeGene otherCompositeGene = (CompositeGene) a_other;
462  if (otherCompositeGene.isEmpty()) {
463  // If our value is also null, then we're the same. Otherwise,
464  // this is the greater gene.
465  // ----------------------------------------------------------
466  if (isEmpty()) {
467  return 0;
468  }
469  else {
470  return 1;
471  }
472  }
473  else {
474  // Compare each gene against each other.
475  // -------------------------------------
476  int numberGenes = Math.min(size(), otherCompositeGene.size());
477  Gene gene1;
478  Gene gene2;
479  for (int i = 0; i < numberGenes; i++) {
480  gene1 = geneAt(i);
481  gene2 = otherCompositeGene.geneAt(i);
482  int result = gene1.compareTo(gene2);
483  if (result != 0) {
484  return result;
485  }
486  }
487  // If everything is equal until now the CompositeGene with more
488  // contained genes wins.
489  // ------------------------------------------------------------
490  if (size() == otherCompositeGene.size()) {
491  if (isCompareApplicationData()) {
492  return compareApplicationData(getApplicationData(),
493  otherCompositeGene.getApplicationData());
494  }
495  else {
496  return 0;
497  }
498  }
499  else {
500  if (size() > otherCompositeGene.size()) {
501  return 1;
502  }
503  else {
504  return -1;
505  }
506  }
507  }

647  StringTokenizer st = new StringTokenizer
