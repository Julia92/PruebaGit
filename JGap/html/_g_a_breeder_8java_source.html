56  public Population evolve(Population a_pop, Configuration a_conf) {
57  Population pop = a_pop;
58  BulkFitnessFunction bulkFunction = a_conf.getBulkFitnessFunction();
59  boolean monitorActive = a_conf.getMonitor() != null;
60  IChromosome fittest = null;
61  // If first generation: Set age to one to allow genetic operations,
62  // see CrossoverOperator for an illustration.
63  // ----------------------------------------------------------------
64  if (a_conf.getGenerationNr() == 0) {
65  int size = pop.size();
66  for (int i = 0; i < size; i++) {
67  IChromosome chrom = pop.getChromosome(i);
68  chrom.increaseAge();
69  }
70  // If a bulk fitness function has been provided, call it.
71  // ------------------------------------------------------
72  if (bulkFunction != null) {
73  try {
74  pop = bulkFunctionEvaluation(a_conf, bulkFunction, pop, monitorActive);
75  } catch (InvalidConfigurationException ex) {
76  throw new RuntimeException(ex);
77  }
78  // Increase number of generations.
79  // -------------------------------
80  a_conf.incrementGenerationNr();
81  // Fire an event to indicate we've performed an evolution.
82  // -------------------------------------------------------
83  m_lastPop = pop;
84  m_lastConf = a_conf;
85  a_conf.getEventManager().fireGeneticEvent(
86  new GeneticEvent(GeneticEvent.GENOTYPE_EVOLVED_EVENT, this));
87  return pop;
88  }
89  }
90  else {
91  // Select fittest chromosome in case it should be preserved and we are
92  // not in the very first generation.
93  // -------------------------------------------------------------------
94  if (a_conf.isPreserveFittestIndividual()) {
97  fittest = pop.determineFittestChromosome(0, pop.size() - 1);
98  }
99  }
100  if (a_conf.getGenerationNr() > 0 && bulkFunction == null) {
101  // Adjust population size to configured size (if wanted).
102  // Theoretically, this should be done at the end of this method.
103  // But for optimization issues it is not. If it is the last call to
104  // evolve() then the resulting population possibly contains more
105  // chromosomes than the wanted number. But this is no bad thing as
106  // more alternatives mean better chances having a fit candidate.
107  // If it is not the last call to evolve() then the next call will
108  // ensure the correct population size by calling keepPopSizeConstant.
109  // ------------------------------------------------------------------
110  keepPopSizeConstant(pop, a_conf);
111  }
112  // Ensure fitness value of all chromosomes is udpated.
113  // ---------------------------------------------------
114  if (monitorActive) {
115  // Monitor that fitness value of chromosomes is being updated.
116  // -----------------------------------------------------------
117  a_conf.getMonitor().event(
118  IEvolutionMonitor.MONITOR_EVENT_BEFORE_UPDATE_CHROMOSOMES1,
119  a_conf.getGenerationNr(), new Object[] {pop});
120  }
121  updateChromosomes(pop, a_conf);
122  if (monitorActive) {
123  // Monitor that fitness value of chromosomes is being updated.
124  // -----------------------------------------------------------
125  a_conf.getMonitor().event(
126  IEvolutionMonitor.MONITOR_EVENT_AFTER_UPDATE_CHROMOSOMES1,
127  a_conf.getGenerationNr(), new Object[] {pop});
128  }
129  // Apply certain NaturalSelectors before GeneticOperators will be executed.
130  // ------------------------------------------------------------------------
131  pop = applyNaturalSelectors(a_conf, pop, true);
132  int newChromIndex = pop.size();
133  // Execute all of the Genetic Operators.
134  // -------------------------------------
135  applyGeneticOperators(a_conf, pop);
136  // Reset fitness value of genetically operated chromosomes.
137  // Normally, this should not be necessary as the Chromosome class
138  // initializes each newly created chromosome with
139  // FitnessFunction.NO_FITNESS_VALUE. But who knows which Chromosome
140  // implementation is used or if cloning is utilized.
141  // ----------------------------------------------------------------
142  int currentPopSize = pop.size();
143  for (int i = newChromIndex; i < currentPopSize; i++) {
144  IChromosome chrom = pop.getChromosome(i);
145  chrom.setFitnessValueDirectly(FitnessFunction.NO_FITNESS_VALUE);
146  // Mark chromosome as new-born.
147  // ----------------------------
148  chrom.resetAge();
149  // Mark chromosome as being operated on.
150  // -------------------------------------
151  chrom.increaseOperatedOn();
152  }
153  // Increase age of all chromosomes which are not modified by genetic
154  // operations.
155  // -----------------------------------------------------------------
156  int size = Math.min(newChromIndex, currentPopSize);
157  for (int i = 0; i < size; i++) {
158  IChromosome chrom = pop.getChromosome(i);
159  chrom.increaseAge();
160  // Mark chromosome as not being operated on.
161  // -----------------------------------------
162  chrom.resetOperatedOn();
163  }
164  // If a bulk fitness function has been provided, call it.
165  // ------------------------------------------------------
166  if (bulkFunction != null & a_conf.getGenerationNr() > 0) {
167  try {
168  pop = bulkFunctionEvaluation(a_conf, bulkFunction, pop, monitorActive);
169  } catch (InvalidConfigurationException ex) {
170  throw new RuntimeException(ex);
171  }
172  }
173  // Ensure fitness value of all chromosomes is udpated.
174  // ---------------------------------------------------
175  if (monitorActive) {
176  // Monitor that fitness value of chromosomes is being updated.
177  // -----------------------------------------------------------
178  a_conf.getMonitor().event(
179  IEvolutionMonitor.MONITOR_EVENT_BEFORE_UPDATE_CHROMOSOMES2,
180  a_conf.getGenerationNr(), new Object[] {pop});
181  }
182  updateChromosomes(pop, a_conf);
183  if (monitorActive) {
184  // Monitor that fitness value of chromosomes is being updated.
185  // -----------------------------------------------------------
186  a_conf.getMonitor().event(
187  IEvolutionMonitor.MONITOR_EVENT_AFTER_UPDATE_CHROMOSOMES2,
188  a_conf.getGenerationNr(), new Object[] {pop});
189  }
190  // Apply certain NaturalSelectors after GeneticOperators have been applied.
191  // ------------------------------------------------------------------------
192  pop = applyNaturalSelectors(a_conf, pop, false);
193  // Fill up population randomly if size dropped below specified percentage
194  // of original size.
195  // ----------------------------------------------------------------------
196  fillPopulationRandomlyToOriginalSize(a_conf, pop);
197  IChromosome newFittest = reAddFittest(pop, fittest);
198  if (monitorActive && newFittest != null) {
199  // Monitor that fitness value of chromosomes is being updated.
200  // -----------------------------------------------------------
201  a_conf.getMonitor().event(

203  a_conf.getGenerationNr(), new Object[] {pop, fittest});
204  }

206  // -------------------------------
207  a_conf.incrementGenerationNr();
208  // Fire an event to indicate we've performed an evolution.

210  m_lastPop = pop;

213  new GeneticEvent(GeneticEvent.GENOTYPE_EVOLVED_EVENT, this));

217  private void fillPopulationRandomlyToOriginalSize(Configuration a_conf,
218  Population pop) {
219  boolean monitorActive = a_conf.getMonitor() != null;
220  // Fill up population randomly if size dropped below specified percentage
221  // of original size.
222  // ----------------------------------------------------------------------
223  if (a_conf.getMinimumPopSizePercent() > 0) {
224  int sizeWanted = a_conf.getPopulationSize();
225  int popSize;
226  int minSize = (int) Math.round(sizeWanted *
227  (double) a_conf.getMinimumPopSizePercent()
228  / 100);
229  popSize = pop.size();
230  if (popSize < minSize) {
231  IChromosome newChrom;
232  IChromosome sampleChrom = a_conf.getSampleChromosome();
233  Class sampleChromClass = sampleChrom.getClass();
234  IInitializer chromIniter = a_conf.getJGAPFactory().
235  getInitializerFor(sampleChrom, sampleChromClass);
236  while (pop.size() < minSize) {
237  try {
240  newChrom = (IChromosome) chromIniter.perform(sampleChrom,
241  sampleChromClass, null);
242  if (monitorActive) {
243  // Monitor that fitness value of chromosomes is being updated.
244  // -----------------------------------------------------------
245  a_conf.getMonitor().event(
246  IEvolutionMonitor.MONITOR_EVENT_BEFORE_ADD_CHROMOSOME,
247  a_conf.getGenerationNr(), new Object[] {pop, newChrom});
248  }
249  pop.addChromosome(newChrom);
250  } catch (Exception ex) {
251  throw new RuntimeException(ex);
252  }
253  }
254  }
255  }

349  public Population bulkFunctionEvaluation(Configuration a_conf,
350  BulkFitnessFunction a_bulkFunction, Population a_pop,
351  boolean a_monitorActive)
352  throws InvalidConfigurationException {
353  if (a_bulkFunction != null) {
354  if (a_monitorActive) {
355  // Monitor that bulk fitness will be called for evaluation.
356  // --------------------------------------------------------
357  a_conf.getMonitor().event(
358  IEvolutionMonitor.MONITOR_EVENT_BEFORE_BULK_EVAL,
359  a_conf.getGenerationNr(), new Object[] {a_bulkFunction, a_pop});
360  }
361  //remove chromosomes which have been already evaluated
362  Population popForBulkFunction = removeEvaluatedChromosomes(a_pop, a_conf);
363  if (!m_allChromosomesSoFar.isEmpty()) {
364  Iterator it = popForBulkFunction.getChromosomes().iterator();
365  //remove dublicates
366  while (it.hasNext()) {
367  IChromosome a_chrom1 = (IChromosome) it.next();
368  if (m_allChromosomesSoFar.contains( (IChromosome) a_chrom1)) {
369  it.remove();
370  }
371  }
372  }
373  fillPopulationRandomlyToOriginalSize(a_conf, popForBulkFunction);
374  //When we call it here, it will remove non evaluated chromosomes
375  keepPopSizeConstant(popForBulkFunction, a_conf);
376  if (popForBulkFunction.size() > 0) {
379  a_bulkFunction.evaluate(popForBulkFunction);
380  }
381  //Remove the fitness value and add evaluated elements to
382  //the List with NO_FITNESS_VALUE
383  //and add the NEW elements to pop
384  Iterator it2 = popForBulkFunction.getChromosomes().iterator();
385  while (it2.hasNext()) {
386  Chromosome chrom0 = (Chromosome) it2.next();
387  Chromosome chrom = (Chromosome) chrom0.clone();
388  chrom.setFitnessValueDirectly(FitnessFunction.NO_FITNESS_VALUE);
389  m_allChromosomesSoFar.add(chrom);
390  if (!a_pop.getChromosomes().contains( (IChromosome) chrom0)) {
391  a_pop.addChromosome( (Chromosome) chrom0);
392  }
393  }
394  //remove chromosomes which have been already evaluated
395  a_pop = removeChromosomesWithoutFitnessValue(a_pop, a_conf);
396  if (a_monitorActive) {
397  // Monitor that bulk fitness has been called for evaluation.
398  // ---------------------------------------------------------
399  a_conf.getMonitor().event(
400  IEvolutionMonitor.MONITOR_EVENT_AFTER_BULK_EVAL,
401  a_conf.getGenerationNr(), new Object[] {a_bulkFunction, a_pop});
402  }
403  }
404  return a_pop;
405  }
