137  public TreeNode createTree(IGPProgram a_prog)
138  throws InvalidConfigurationException {
139  if (a_prog == null) {
140  return null;
141  }
142  ProgramChromosome master = new ProgramChromosome(m_conf);
143  master.setIndividual(a_prog);
144  TreeNode tree;
145  if (a_prog.size() > 1) {
146  Class[] types = new Class[a_prog.size()];
147  for (int i = 0; i < a_prog.size(); i++) {
148  types[i] = CommandGene.VoidClass; //this is arbitrary
149  }
150  master.setGene(0, new SubProgram(m_conf, types));
151  int index = 1;
152  for (int i = 0; i < a_prog.size(); i++) {
153  ProgramChromosome child = a_prog.getChromosome(i);
154  for (int j = 0; j < child.size(); j++) {
155  master.setGene(index++, child.getGene(j));
156  }
157  }
158  master.redepth();
159  tree = new JGAPTreeNode(master, 0);
160  }
161  else {
162  tree = new JGAPTreeNode(a_prog.getChromosome(0), 0);
163  }
164  return tree;
165  }
