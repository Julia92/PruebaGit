274  public void create(Class[] a_types, Class[][] a_argTypes,
275  CommandGene[][] a_nodeSets, int[] a_minDepths,
276  int[] a_maxDepths, int a_maxNodes,
277  boolean[] a_fullModeAllowed,
278  IProgramCreator a_programCreator,
279  int a_offset)
280  throws InvalidConfigurationException {
281  int divisor;
282  if (m_popSize < 2) {
283  divisor = 1;
284  }
285  else {
286  divisor = m_popSize - 1;
287  }
288  int genNr = getGPConfiguration().getGenerationNr();
289  int genI = new Random().nextInt(m_popSize);
290  RandomGenerator generator = getGPConfiguration().getRandomGenerator();
291  int minDepth = getGPConfiguration().getMinInitDepth();
292  int maxDepth = getGPConfiguration().getMaxInitDepth();
293  for (int i = a_offset; i < m_popSize; i++) {
294  IGPProgram program = null;
295  // Vary depth dependent on run index.
296  // ----------------------------------
298  int depth = minDepth
299  + (maxDepth - minDepth) * i
300  / divisor;
301 
302  // Create new GP program.
303  // ----------------------
304  int tries = 0;
305  int maxTries = getGPConfiguration().getProgramCreationMaxtries();
306  do {
307  try {
308  // Randomize grow option as growing produces a valid program
309  // more likely than the full mode.
310  // ---------------------------------------------------------
311  boolean grow;
312  if (i % 2 == 0 || generator.nextInt(8) > 6) {
313  grow = true;
314  }
315  else {
316  grow = false;
317  }
318  program = create(i, a_types, a_argTypes, a_nodeSets,
319  a_minDepths, a_maxDepths, depth, grow,
320  a_maxNodes, a_fullModeAllowed, tries,
321  a_programCreator);
322  if (program != null && getGPConfiguration().getPrototypeProgram() == null) {
323  // Remember a prototype of a valid program in case generation
324  // cannot find a valid program within some few tries
325  // --> then clone the prototype.
326  // Necessary if the maxNodes parameter is chosen small or a
327  // validator is used which is quite restrictive.
328  // ----------------------------------------------------------
329  getGPConfiguration().setPrototypeProgram(program);
330  LOGGER.info("Prototype program set");
331  }
332  else if (genNr % 5 == 0 && genNr > 0 && i == genI) {
334  // Set prototype to new value after every some evolutions.
335  // -------------------------------------------------------
336  double protoFitness = getGPConfiguration().getPrototypeProgram().
337  getFitnessValue();
338  if (getGPConfiguration().getGPFitnessEvaluator().isFitter(program.
339  getFitnessValue(), protoFitness)) {
340  getGPConfiguration().setPrototypeProgram(program);
341  }
342  }
343  break;
344  } catch (IllegalStateException iex) {
345  if (depth < maxDepth) {
346  depth = depth + generator.nextInt(2);
347  }
348  else {
349  depth = depth - generator.nextInt(4);
350  if (depth < minDepth) {
351  depth = minDepth;
352  }
353  }
354  tries++;
355  if (maxTries >=0 && tries > maxTries || (i > a_offset && tries > 40)) {
356  IGPProgram prototype = getGPConfiguration().getPrototypeProgram();
357  if (prototype != null) {
358  ICloneHandler cloner = getGPConfiguration().getJGAPFactory().
359  getCloneHandlerFor(prototype, null);
360  if (cloner != null) {
361  try {
362  program = (IGPProgram) cloner.perform(prototype, null, null);
363  if (warningPrototypeReused > 0) {
364  // Only output once.
365  // -----------------
366  LOGGER.warn("Prototype program reused because random"
367  + " program did not satisfy constraints");
368  warningPrototypeReused++;
369  }
370  break;
371  } catch (Exception ex) {
372  // Rethrow original error.
373  // -----------------------
374  throw iex;
375  }
376  }
377  else {
378  LOGGER.warn("Warning: no clone handler found for"
379  + " prototype program type "
380  + prototype);
381  }
382  }
383  // Rethrow original error.
384  // -----------------------
385  throw iex;
386  }
387  }
388  } while (true);
389  setGPProgram(i, program);
390  }
391  setChanged(true);

657  public IGPProgram determineFittestProgram() {
658  if (!m_changed && m_fittestProgram != null) {
659  return m_fittestProgram;
660  }
661  double bestFitness = -1.0d;
662  IGPFitnessEvaluator evaluator = getGPConfiguration().getGPFitnessEvaluator();
663  double fitness;
664  for (int i = 0; i < m_programs.length && m_programs[i] != null; i++) {
665  IGPProgram program = m_programs[i];
666  try {
667  fitness = program.getFitnessValue();
668  } catch (IllegalStateException iex) {
669  fitness = Double.NaN;
670  }
671  if (!Double.isNaN(fitness) &&
672  Math.abs(GPFitnessFunction.NO_FITNESS_VALUE - fitness) > DELTA) {
673  if (m_fittestProgram == null || evaluator.isFitter(fitness, bestFitness)) {
674  bestFitness = fitness;
675  m_fittestProgram = program;
676  }
677  }
678  }
679  setChanged(false);
680  if (m_fittestProgram != null) {
681  IJGAPFactory factory = getGPConfiguration().getJGAPFactory();
682  if (factory == null) {

684  }

686  if (cloner != null) {

688  m_fittestProgram = (IGPProgram) cloner.perform(m_fittestProgram, null, null);
689  } catch (Exception ex) {

692  }

706  public IGPProgram determineFittestProgramComputed() {
707  double bestFitness = -1.0d;
708  IGPFitnessEvaluator evaluator = getGPConfiguration().getGPFitnessEvaluator();
709  double fitness;
710  IGPProgram fittest = null;
711  for (int i = 0; i < m_programs.length && m_programs[i] != null; i++) {
712  IGPProgram program = m_programs[i];
713  if (program instanceof GPProgramBase) {
714  GPProgramBase program1 = (GPProgramBase) program;
715  fitness = program1.getFitnessValueDirectly();
716  }
717  else {
718  fitness = program.getFitnessValue();
719  }
720  if (Math.abs(fitness - FitnessFunction.NO_FITNESS_VALUE) >
721  FitnessFunction.DELTA) {
722  if (fittest == null || evaluator.isFitter(fitness, bestFitness)) {
723  fittest = program;
724  bestFitness = fitness;
725  }
726  }
727  }
728  return fittest;
