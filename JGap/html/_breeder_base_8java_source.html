49  protected Population applyNaturalSelectors(Configuration a_config,
50  Population a_pop, boolean a_processBeforeGeneticOperators) {
52  boolean monitorActive = a_config.getMonitor() != null;
53  try {
54  int selectorSize = a_config.getNaturalSelectorsSize(
55  a_processBeforeGeneticOperators);
56  if (selectorSize > 0) {
57  int population_size = a_config.getPopulationSize();
58  // Only select part of the previous population into this generation.
59  // -----------------------------------------------------------------
60  population_size = (int) Math.round(population_size *
61  a_config.getSelectFromPrevGen());
62  int single_selection_size;
63  Population new_population = new Population(a_config, population_size);
64  NaturalSelector selector;
65  // Repopulate the population of chromosomes with those selected
66  // by the natural selector. Iterate over all natural selectors.
67  // ------------------------------------------------------------
68  for (int i = 0; i < selectorSize; i++) {
69  selector = a_config.getNaturalSelector(
70  a_processBeforeGeneticOperators, i);
71  if (i == selectorSize - 1 && i > 0) {
72  // Ensure the last NaturalSelector adds the remaining Chromosomes.
73  // ---------------------------------------------------------------
74  single_selection_size = population_size - new_population.size();
75  }
76  else {
77  single_selection_size = population_size / selectorSize;
78  }
79  if (monitorActive) {
80  // Monitor that selection is going to be performed.
81  // ------------------------------------------------
82  a_config.getMonitor().event(
83  IEvolutionMonitor.MONITOR_EVENT_BEFORE_SELECT,
84  a_config.getGenerationNr(),
85  new Object[] {selector, a_pop, single_selection_size,
86  a_processBeforeGeneticOperators});
87  }
88  // Do selection of chromosomes.
89  // ----------------------------
91  selector.select(single_selection_size, a_pop, new_population);
92  if (monitorActive) {
93  // Monitor population after selection took place.
94  // ----------------------------------------------
95  a_config.getMonitor().event(
96  IEvolutionMonitor.MONITOR_EVENT_AFTER_SELECT,
97  a_config.getGenerationNr(),
98  new Object[] {selector, a_pop, new_population,
99  single_selection_size, a_processBeforeGeneticOperators});
100  }
101  // Clean up the natural selector.
102  // ------------------------------
103  selector.empty();
104  }
105  return new_population;
106  }
107  else {
108  return a_pop;
109  }
110  } catch (InvalidConfigurationException iex) {
111  // This exception should never be reached.
112  // ---------------------------------------
113  throw new IllegalStateException(iex);
114  }
115  }

130  while (operatorIterator.hasNext()) {
131  GeneticOperator operator = (GeneticOperator) operatorIterator.next();
133  // Fire listener before genetic operator will be executed.
134  // -------------------------------------------------------
135  a_config.getEventManager().fireGeneticEvent(
136  new GeneticEvent(GeneticEvent.BEFORE_GENETIC_OPERATOR, new Object[] {
137  this, operator}));
138  if (monitorActive) {
139  // Monitor that operator will be performed.
140  // ----------------------------------------
141  a_config.getMonitor().event(
142  IEvolutionMonitor.MONITOR_EVENT_BEFORE_OPERATE,
143  a_config.getGenerationNr(),
144  new Object[] {operator, a_pop, a_pop.getChromosomes()});
145  }
146  operator.operate(a_pop, a_pop.getChromosomes());
147  if (monitorActive) {

152  a_config.getGenerationNr(),

154  }

156  // -------------------------------------------------------
157  a_config.getEventManager().fireGeneticEvent(
