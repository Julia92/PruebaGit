184  public void execute_void(ProgramChromosome c, int n, Object[] args) {
185  // Determine the end index of the loop (child at index 0).
186  // -------------------------------------------------------
187  int x;
188  if (m_endIndex == -1) {
189  if (m_typeVar == CommandGene.IntegerClass) {
190  x = c.execute_int(n, 0, args);
191  }
192  else if (m_typeVar == CommandGene.LongClass) {
193  x = (int) c.execute_long(n, 0, args);
194  }
195  else if (m_typeVar == CommandGene.DoubleClass) {
196  x = (int) Math.round(c.execute_double(n, 0, args));
197  }
198  else if (m_typeVar == CommandGene.FloatClass) {
199  x = (int) Math.round(c.execute_float(n, 0, args));
200  }
201  else {
202  throw new RuntimeException("Type "
203  + m_typeVar
204  + " not supported by ForLoop");
205  }
206  if (x > m_maxLoop) {
207  x = m_maxLoop;
208  }
209  // Repeatedly execute the second child (index = 1).
210  // ------------------------------------------------
211  for (int i = m_startIndex; i < x; i++) {
212  c.execute_void(n, 1, args);
213  }
214  }
215  else {
216  // Repeatedly execute the first child (index = 0).
217  // -----------------------------------------------
218  for (int i = m_startIndex; i < m_endIndex; i = i + m_increment) {
219  // Store counter in memory.
220  // ------------------------
221  getGPConfiguration().storeInMemory(ForLoop.INTERNAL_COUNTER_STORAGE,
222  new Integer(i));
223  c.execute_void(n, 0, args);
224  }
225  }
