269  private int round(double value, Integer factor) {
270  if (value % factor == 0) {
271  return (int) value;
272  }
273  int floor = (int) ( (value / factor)) * factor;
274  int ceiling = floor + factor;
275  if (ceiling - value <= value - floor && ceiling <= m_upperBounds) {
276  return ceiling;
277  }
278  else if (floor >= m_lowerBounds) {
279  return floor;
280  }
281  else {
282  return ceiling;

315  protected void mapValueToWithinBounds() {
316  if (getAllele() != null) {
317  Integer i_value = ( (Integer) getAllele());
318  // If the value exceeds either the upper or lower bounds, then
319  // map the value to within the legal range. To do this, we basically
320  // calculate the distance between the value and the integer min,
321  // determine how many bounds units that represents, and then add
322  // that number of units to the upper bound.
323  // -----------------------------------------------------------------
324  if (i_value.intValue() > m_upperBounds ||
325  i_value.intValue() < m_lowerBounds) {
326  RandomGenerator rn;
327  if (getConfiguration() != null) {
328  rn = getConfiguration().getRandomGenerator();

334  setAllele(new Integer(m_lowerBounds));
