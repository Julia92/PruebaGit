135  trackChromosomes(Map<Integer, Map> a_data, Population a_pop) {
136  // A track is a list of chromosomes, with each of them having a list
137  // of originators:

139  // -------------------------------------------------------------------------
140  Map<IChromosome, Map<Integer, List<IChromosome>>> chromosomesTrack

142  int evNo;
143  int currentIndex = evolutionIndexes.size();
144  if (currentIndex < 1) {
145  return chromosomesTrack;
146  }
147  // Retrieve predecessing population.
148  // ---------------------------------
149  evNo = evolutionIndexes.get(currentIndex - 1);
150  Map<Integer, PopulationContext> contextMap = a_data.get(evNo);

152  CONTEXT_END_OF_CYCLE);

154  Map<Integer, List<IChromosome>> evoMap = new Hashtable();
155  // Find originators for all chromosomes in given population.
156  // ---------------------------------------------------------

158  IChromosome chrom = a_pop.getChromosome(i);
159  int uindex = 1;
160  List<IChromosome> uids = new Vector();
161  // Gather originators for current chromosome in population.
162  // --------------------------------------------------------
163  do {
164  String uid = chrom.getUniqueIDTemplate(uindex++);
165  IChromosome chrom0 = null;
166  if (uid != null || uindex == 2) {
167  // Find chromosome instance for unique ID.
168  // ---------------------------------------
169  for (int k = 0; k < previousPop.size(); k++) {
