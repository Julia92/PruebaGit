254  // ----------------------------------------------------------------------
255  CommandGene n;

490  protected CommandGene selectNode(int a_chromIndex, Class a_returnType,
491  int a_subReturnType,
492  CommandGene[] a_functionSet,
493  boolean a_function, boolean a_growing) {
494  // Determine possible functions.
495  // -----------------------------
496  Vector<CommandGene> possibleFunctions=new Vector<CommandGene>(0);
497  IGPProgram ind = getIndividual();
498  ISingleNodeValidator singleNodeValidator = getGPConfiguration().
499  getSingleNodeValidator();
500  for (int i = 0; i < a_functionSet.length; i++) {
501  if (a_functionSet[i].getReturnType() == a_returnType
502  && (a_subReturnType == 0
503  || a_subReturnType == a_functionSet[i].getSubReturnType())) {
504  if (a_functionSet[i].getArity(ind) == 0 && (!a_function || a_growing)) {
505  // Verify if function/terminal is allowed here.
506  // --------------------------------------------
507  if (singleNodeValidator == null ||
508  singleNodeValidator.isAllowed(a_chromIndex, this, a_functionSet,
509  a_functionSet[i], a_returnType, a_subReturnType, m_index)) {
510  possibleFunctions.add(a_functionSet[i]);
511  }
512  }
513  if (a_functionSet[i].getArity(ind) != 0 && a_function) {
514  // Verify if function/terminal is allowed here.
515  // --------------------------------------------
516  if (singleNodeValidator == null ||
517  singleNodeValidator.isAllowed(a_chromIndex, this, a_functionSet,
518  a_functionSet[i],
519  a_returnType, a_subReturnType, m_index)) {
520  possibleFunctions.add(a_functionSet[i]);
521  }
522  }
523  }
524  }
525  // Error handing in case no valid function found.
526  // ----------------------------------------------
527  if (possibleFunctions.isEmpty()) {
528  if (a_growing && (a_returnType == CommandGene.VoidClass
529  || a_returnType == Void.class)) {
530  // We simply return a NOP, it does nothing :-)
531  // -------------------------------------------
532  try {
533  return new NOP(getGPConfiguration(), a_subReturnType);
534  } catch (InvalidConfigurationException iex) {
535  // Should never happen.
536  // --------------------
537  throw new RuntimeException(iex);
538  }
539  }
540  final String errormsg = "Chromosome (depth "
541  + getDepth(0)
542  + ", index " + a_chromIndex
543  + ") requires a " +
544  (a_function ?
545  ("function" + (a_growing ? " or terminal" : ""))
546  : "terminal") + " of return type " +
547  a_returnType
548  + " (sub return type " + a_subReturnType + ")"
549  + " but there is no such node available";
550  if (!getGPConfiguration().isStrictProgramCreation()) {
551  // Allow another try in case it is allowed.
552  // ----------------------------------------
553  throw new IllegalStateException(errormsg);
554  }
555  else {
556  // Interrupt the whole evolution process.
557  // --------------------------------------
558  throw new RuntimeException(errormsg);
559  }
560  }
561  // Select a function randomly.
562  // ---------------------------
563  int index = getGPConfiguration().getRandomGenerator().nextInt(
564  possibleFunctions.size());
565  CommandGene n = possibleFunctions.elementAt(index);
566  return n;
567  }

591  protected CommandGene[] growOrFullNode(int a_num, int a_depth,
592  Class a_returnType, int a_subReturnType, CommandGene[] a_functionSet,
593  CommandGene a_rootNode, int a_recurseLevel,
594  boolean a_grow, int a_childNum, boolean a_validateNode) {
595  boolean mutated = false;
596  boolean uncloned = true;
597  GPConfiguration conf = getGPConfiguration();
598  RandomGenerator random = conf.getRandomGenerator();
599  if (a_rootNode == null || a_validateNode) {
600  int tries = 0;
601  int evolutionRound = getGPConfiguration().getGenerationNr();
602  boolean aFunction = a_depth >= 1;
603  // Clone the array, not the content of the array.
604  // ----------------------------------------------
605  CommandGene[] localFunctionSet = (CommandGene[]) a_functionSet.clone();
606  int len = a_functionSet.length;
607  do {
608  CommandGene node = selectNode(a_num, a_returnType, a_subReturnType,
609  localFunctionSet, aFunction, a_grow);
610  if (!conf.validateNode(this, node, a_rootNode, tries++, a_num,
611  a_recurseLevel, a_returnType, localFunctionSet,
612  a_depth, a_grow, a_childNum, false)) {
613  // In the first round of evolution ensure to always have one valid
614  // individual as we need a prototype for cloning!
615  // ---------------------------------------------------------------
616  if (evolutionRound > 0 || tries <= len*2) {
617  // Remove invalid node from local function set.
618  // --------------------------------------------
619  localFunctionSet = remove(localFunctionSet, node);
620  if (localFunctionSet.length == 0) {
621  throw new IllegalStateException("No appropriate function found"
622  + " during program creation!");
623  }
624  continue;
625  }
626  }
627  // Optionally use a mutant/clone of the originally selected command
628  // instead of reusing the same command instance.
629  // ----------------------------------------------------------------
630  if (random.nextDouble() <= conf.getMutationProb()) {
631  if (IMutateable.class.isAssignableFrom(node.getClass())) {
632  try {
633  CommandGene node2 = ( (IMutateable) node).applyMutation(0,
634  random.nextDouble());
635  // Check if mutant's function is allowed.
636  // --------------------------------------
637  if (getCommandOfClass(0, node2.getClass()) >= 0) {
638  mutated = true;
639  if (node2 != node) {
640  node = node2;
641  uncloned = false;
642  }
643  }
644  } catch (InvalidConfigurationException iex) {
645  // Ignore but log.
646  // ---------------
647  LOGGER.warn("Ignored problem", iex);
648  }
649  }
650  }
651  // Avoid using commands more than once if allowed only once.
652  // ---------------------------------------------------------
653  if (IUniqueCommand.class.isAssignableFrom(node.getClass())) {
654  a_functionSet = remove(a_functionSet, node);
655  }
656  a_rootNode = node;
657  break;
658  } while (true);
659  }
660  // Generate the new node.
661  // ----------------------
662  m_depth[m_index] = m_maxDepth - a_depth;
663  // Optional dynamize the arity for commands with a flexible number
664  // of children. Normally, dynamizeArity does nothing, see declaration
665  // of method in CommandGene, which can be overridden in sub classes.
666  // -------------------------------------------------------------------
667  boolean dynamize = random.nextDouble() <= conf.getDynamizeArityProb();
668  if (dynamize) {
669  a_rootNode.dynamizeArity();
670  }
671  // Clone node if possible and not already done via mutation.
672  // ---------------------------------------------------------
673  if (uncloned && !conf.isNoCommandGeneCloning()
674  && a_rootNode instanceof ICloneable) {
676  a_rootNode = (CommandGene) ( (ICloneable) a_rootNode).clone();
677  m_genes[m_index++] = a_rootNode;
678  }
679  else {
680  m_genes[m_index++] = a_rootNode;
681  }
682  if (a_depth >= 1) {
683  IGPProgram ind = getIndividual();
684  int arity = a_rootNode.getArity(ind);
685  for (int i = 0; i < arity; i++) {
686  // Ensure required depth is cared about.
687  // -------------------------------------
688  if (m_index < m_depth.length) {
689  a_functionSet = growOrFullNode(a_num, a_depth - 1,
690  a_rootNode.getChildType(getIndividual(), i),
691  a_rootNode.getSubChildType(i),
692  a_functionSet, a_rootNode, a_recurseLevel + 1, a_grow,
693  i, true);
694  }
695  else {
696  // No valid program could be generated. Abort.
697  // -------------------------------------------
698  throw new IllegalStateException("Randomly created program violates"
699  + " configuration constraints (symptom 1). It may be that you"
700  + " specified a too small number of maxNodes to use"
701  + " (current arity: "
702  + i
703  + ", overall arity: "
704  + arity
705  + ")!");
706  }
707  }
708  }
709  else {
710  if (a_rootNode.getArity(getIndividual()) > 0) {
711  // No valid program could be generated. Abort.
712  // -------------------------------------------
713  throw new IllegalStateException("Randomly created program violates"
714  + " configuration constraints"
715  + " (symptom 2):"
716  + " Root node: "
717  + a_rootNode.getClass().toString());
718  }
719  }
720  return a_functionSet;
721  }

1200  public int compareTo(Object a_other) {
1201  // First, if the other Chromosome is null, then this chromosome is
1202  // automatically the "greater" Chromosome.
1203  // ---------------------------------------------------------------
1204  if (a_other == null) {
1205  return 1;
1206  }
1207  int size = size();
1208  ProgramChromosome otherChromosome = (ProgramChromosome) a_other;
1209  CommandGene[] otherGenes = otherChromosome.m_genes;
1210  // If the other Chromosome doesn't have the same number of genes,
1211  // then whichever has more is the "greater" Chromosome.
1212  // --------------------------------------------------------------
1213  if (otherChromosome.size() != size) {
1214  return size() - otherChromosome.size();
1215  }
1216  // Next, compare the gene values (alleles) for differences. If
1217  // one of the genes is not equal, then we return the result of its
1218  // comparison.
1219  // ---------------------------------------------------------------
1220  for (int i = 0; i < size; i++) {
1221  int comparison = m_genes[i].compareTo(otherGenes[i]);
1222  if (comparison != 0) {
1223  return comparison;
1224  }
1225  }
1227  if (isCompareApplicationData()) {
1228  // Compare application data.
1229  // -------------------------
1230  if (getApplicationData() == null) {
1231  if (otherChromosome.getApplicationData() != null) {
1232  return -1;
1233  }
1234  }
1235  else if (otherChromosome.getApplicationData() == null) {
1236  return 1;
1237  }
1238  else {
1239  if (getApplicationData() instanceof Comparable) {
1240  try {
1241  return ( (Comparable) getApplicationData()).compareTo(
1242  otherChromosome.getApplicationData());
1243  } catch (ClassCastException cex) {
1244  return -1;
1245  }
1246  }
1247  else {
1248  return getApplicationData().getClass().getName().compareTo(
1249  otherChromosome.getApplicationData().getClass().getName());
1250  }
1251  }
1252  }
1253  // Everything is equal. Return zero.
1254  // ---------------------------------
1255  return 0;
1256  }
