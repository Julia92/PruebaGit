147  private void parseLine(String a_line)
148  throws MetaConfigException {
149  String[] tokens = a_line.split("=");
150  if (tokens == null || tokens.length != 2)
151  throw new MetaConfigException(CN + ".parseLine():Exception while " +
152  "parsing " + METACON_FILENAME + " line " +
153  a_line + " is invalid");
154  if (m_state == MetaConfig.INIT && tokens[0].equals("class")) {
155  handleClass(tokens[1]);

157  else if (m_state == MetaConfig.CLASS && tokens[0].equals("property")) {
158  handleProperty(tokens[1]);

160  else if (m_state == MetaConfig.PROPERTY && tokens[0].equals("values")) {
161  handleValues(tokens[1]);

164  handleClass(tokens[1]);
