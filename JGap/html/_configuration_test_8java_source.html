1070  String result = s1;
1071  int index = 0;
1072  while (index < result.length()) {
1073  if (result.charAt(index) == '\n') {
1074  result = result.substring(0, index) + result.substring(index + 1);
1075  }
1076  else {
1077  if (result.charAt(index) == ':') {
1078  if (index < result.length() - 1) {
1079  if (result.charAt(index + 1) == ' ' ||
1080  result.charAt(index + 1) == '\n') {
1081  result = result.substring(0, index + 1) +
1082  result.substring(index + 2);
1083  }
1084  else {
1085  index++;
1086  }
1087  }
1088  else {
1089  index++;
1090  }
1091  }
1092  else if (result.charAt(index) == ' ') {
1093  if (index == 0) {
1094  result = result.substring(1);
1095  }
1096  else if (index == result.length() - 1) {
1097  result = result.substring(0, result.length() - 1);
1098  }
1099  else
1100  if (result.charAt(index + 1) == ' ' ||
1101  result.charAt(index + 1) == '\n') {
1102  result = result.substring(0, index) + result.substring(index + 1);
1103  }
1104  else {
1105  index++;
1106  }
1107  }
1108  else {
1109  index++;
1110  }
1111  }
1112  }
1113  return result;
