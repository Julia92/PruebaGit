285  private int fitnessValue(int[][] antmap, int sandColor, int x, int y) {
286  if (x == antmap[x][y]) {
287  return 0;
288  }
289  else if (x >= antmap[x][y]) {
290  return (x - antmap[x][y]) * (x - antmap[x][y]);
291  }
292  else {
293  //penalty for moving too far to left
294  return (1 + antmap[x][y] - x) * (1 + antmap[x][y] - x);
295  }
296  }
