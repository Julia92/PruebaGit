288  protected void mapValueToWithinBounds() {
289  if (getAllele() != null) {
290  Double d_value = ( (Double) getAllele());
291  if (d_value.isInfinite()) {
292  // Here we have to break to avoid a stack overflow.
293  // ------------------------------------------------
294  return;
295  }
296  // If the value exceeds either the upper or lower bounds, then
297  // map the value to within the legal range. To do this, we basically
298  // calculate the distance between the value and the double min,
299  // then multiply it with a random number and then care that the lower
300  // boundary is added.
301  // ------------------------------------------------------------------
302  if (d_value.doubleValue() > m_upperBound ||
303  d_value.doubleValue() < m_lowerBound) {
304  RandomGenerator rn;
305  if (getConfiguration() != null) {
306  rn = getConfiguration().getRandomGenerator();
307  }
308  else {
309  rn = new StockRandomGenerator();

313  setAllele(new Double((rn.nextDouble()
