109  private Operand mutateOperand(Operand theOp, double theRate) {
110  Operand aReturn = null;
111  if (theOp instanceof ValueOperand) {
112  RandomGenerator aRand = getConfiguration().getRandomGenerator();
113  if (aRand.nextDouble() < theRate) {
114  // the bigger the rate, the more likely this switches from a set value to a
115  // math operation...
116  MathOperator aOp = generateRandomMathOperator(aRand);
117  // we switch this operand to a math operation, but lets keep its value as one
118  // of the operands of the operator
119  aOp.setLeftOperand(theOp);
120  aReturn = aOp;
121  }
122  else {
123  // lets just mutate the value
124  ValueOperand aOp = (ValueOperand) theOp;
125  aOp.setValue(aOp.value() * (1 + (theRate - .5)));
126  aReturn = aOp;
127  }
128  }
129  else if (theOp instanceof MathOperator) {
130  MathOperator aOp = (MathOperator) theOp;
131  RandomGenerator aRand = getConfiguration().getRandomGenerator();
132  if (aRand.nextDouble() < theRate) {
133  // if there's a value operand in the math operation, pull it out,
134  // and we'll use that. otherwise we'll generate a random one.
135  ValueOperand aVal = new GetValueOperandVisitor().find(aOp);
136  if (aVal == null) {
137  aVal = generateRandomValueOperand(aRand);
138  }
139  aReturn = aVal;
140  }
141  else {
142  int aType = aRand.nextInt(3);
143  if (aType == 0) {
144  // switch the math operation used
145  MathOperator aNewOp = generateRandomMathOperator(aRand);
146  aNewOp.setLeftOperand(aOp.getLeftOperand());
147  aNewOp.setRightOperand(aOp.getRightOperand());
148  aReturn = aNewOp;
149  }
150  else if (aType == 1) {

152  aReturn = aOp;
153  }
154  else if (aType == 2) {
155  aOp.setRightOperand(generateRandomOperand(aRand));
156  aReturn = aOp;
157  }
158  }
159  }
