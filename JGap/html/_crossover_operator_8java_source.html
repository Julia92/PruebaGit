323  public void operate(final Population a_population,
324  final List a_candidateChromosomes) {
325  // Work out the number of crossovers that should be performed.
326  // -----------------------------------------------------------
327  int size = Math.min(getConfiguration().getPopulationSize(),
328  a_population.size());
329  int numCrossovers = 0;
330  if (m_crossoverRate >= 0) {
331  numCrossovers = size / m_crossoverRate;
332  }
333  else if (m_crossoverRateCalc != null) {
334  numCrossovers = size / m_crossoverRateCalc.calculateCurrentRate();
335  }
336  else {
337  numCrossovers = (int) (size * m_crossoverRatePercent);
338  }
339  RandomGenerator generator = getConfiguration().getRandomGenerator();
340  IGeneticOperatorConstraint constraint = getConfiguration().
341  getJGAPFactory().getGeneticOperatorConstraint();
342  // For each crossover, grab two random chromosomes, pick a random
343  // locus (gene location), and then swap that gene and all genes
344  // to the "right" (those with greater loci) of that gene between
345  // the two chromosomes.
346  // --------------------------------------------------------------
347  int index1, index2;
348  //
349  for (int i = 0; i < numCrossovers; i++) {
350  index1 = generator.nextInt(size);
351  index2 = generator.nextInt(size);
352  IChromosome chrom1 = a_population.getChromosome(index1);
353  IChromosome chrom2 = a_population.getChromosome(index2);
354  // Verify that crossover is allowed.
355  // ---------------------------------
356  if (!isXoverNewAge() && chrom1.getAge() < 1 && chrom2.getAge() < 1) {
357  // Crossing over two newly created chromosomes is not seen as helpful
358  // here.
359  // ------------------------------------------------------------------
360  continue;
361  }
362  if (constraint != null) {
363  List v = new Vector();
364  v.add(chrom1);
365  v.add(chrom2);
366  if (!constraint.isValid(a_population, v, this)) {
367  // Constraint forbids crossing over.
368  // ---------------------------------
369  continue;
370  }
371  }
372  // Clone the chromosomes.
373  // ----------------------
374  IChromosome firstMate = (IChromosome) chrom1.clone();
375  IChromosome secondMate = (IChromosome) chrom2.clone();
376  // In case monitoring is active, support it.
377  // -----------------------------------------
378  if (m_monitorActive) {
379  firstMate.setUniqueIDTemplate(chrom1.getUniqueID(), 1);
380  firstMate.setUniqueIDTemplate(chrom2.getUniqueID(), 2);
381  secondMate.setUniqueIDTemplate(chrom1.getUniqueID(), 1);
382  secondMate.setUniqueIDTemplate(chrom2.getUniqueID(), 2);
383  }
384  // Cross over the chromosomes.
385  // ---------------------------
386  doCrossover(firstMate, secondMate, a_candidateChromosomes, generator);
387  }

390  protected void doCrossover(IChromosome firstMate, IChromosome secondMate,
391  List a_candidateChromosomes,
392  RandomGenerator generator) {
393  Gene[] firstGenes = firstMate.getGenes();
394  Gene[] secondGenes = secondMate.getGenes();
395  int locus = generator.nextInt(firstGenes.length);
396  // Swap the genes.
397  // ---------------
398  Gene gene1;
399  Gene gene2;
400  Object firstAllele;
401  for (int j = locus; j < firstGenes.length; j++) {
402  // Make a distinction for ICompositeGene for the first gene.
403  // ---------------------------------------------------------
404  int index = 0;
405  if (firstGenes[j] instanceof ICompositeGene) {
406  // Randomly determine gene to be considered.
407  // -----------------------------------------
408  index = generator.nextInt(firstGenes[j].size());
409  gene1 = ( (ICompositeGene) firstGenes[j]).geneAt(index);
410  }
411  else {
412  gene1 = firstGenes[j];
413  }
414  // Make a distinction for the second gene if CompositeGene.
415  // --------------------------------------------------------
416  if (secondGenes[j] instanceof ICompositeGene) {
417  gene2 = ( (ICompositeGene) secondGenes[j]).geneAt(index);
418  }
419  else {
420  gene2 = secondGenes[j];
421  }
422  if (m_monitorActive) {
423  gene1.setUniqueIDTemplate(gene2.getUniqueID(), 1);
424  gene2.setUniqueIDTemplate(gene1.getUniqueID(), 1);
425  }
426  firstAllele = gene1.getAllele();
427  gene1.setAllele(gene2.getAllele());
428  gene2.setAllele(firstAllele);
429  }
430  // Add the modified chromosomes to the candidate pool so that
431  // they'll be considered for natural selection during the next
432  // phase of evolution.
433  // -----------------------------------------------------------
434  a_candidateChromosomes.add(firstMate);
435  a_candidateChromosomes.add(secondMate);
436  }
