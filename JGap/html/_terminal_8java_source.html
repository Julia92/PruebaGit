205  // If percentage is very high: do mutation not relying on
206  // current value but on a random value.
207  // ------------------------------------------------------
208  if (a_percentage > 0.85d) {
209  setRandomValue();
210  }
211  else {
212  Class retType = getReturnType();

216  (float) a_percentage;

218  newValuef = m_value_float -
