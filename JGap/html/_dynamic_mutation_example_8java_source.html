46  public static void makeChangeForAmount(int a_targetChangeAmount)
47  throws Exception {
48  // Start with a DefaultConfiguration, which comes setup with the
49  // most common settings.
50  // -------------------------------------------------------------
51  Configuration conf = new DefaultConfiguration();
52  // Add custom mutation operator
53  conf.getGeneticOperators().clear();
54 // IUniversalRateCalculator mutCalc = new CoinsMutationRateCalc();
55  TwoWayMutationOperator mutOp = new TwoWayMutationOperator(conf, 7);
56  conf.addGeneticOperator(mutOp);
57  conf.addGeneticOperator(new CrossoverOperator(conf));
58  conf.setPreservFittestIndividual(!true);
59  conf.setKeepPopulationSizeConstant(false);
60  // Set the fitness function we want to use, which is our
61  // MinimizingMakeChangeFitnessFunction. We construct it with
62  // the target amount of change passed in to this method.
63  // ---------------------------------------------------------
64  FitnessFunction myFunc =
65  new DynamicMutationFitnessFunction(a_targetChangeAmount);
66 // conf.setFitnessFunction(myFunc);
67  conf.setBulkFitnessFunction(new BulkFitnessOffsetRemover(myFunc));
68  conf.reset();
69  conf.setFitnessEvaluator(new DeltaFitnessEvaluator());
70  // Now we need to tell the Configuration object how we want our
71  // Chromosomes to be setup. We do that by actually creating a
72  // sample Chromosome and then setting it on the Configuration
73  // object. As mentioned earlier, we want our Chromosomes to each
74  // have four genes, one for each of the coin types. We want the
75  // values (alleles) of those genes to be integers, which represent
76  // how many coins of that type we have. We therefore use the
77  // IntegerGene class to represent each of the genes. That class
78  // also lets us specify a lower and upper bound, which we set
79  // to sensible values for each coin type.
80  // --------------------------------------------------------------
81  Gene[] sampleGenes = new Gene[4];
82  sampleGenes[0] = new IntegerGene(conf, 0, 3 * 10); // Quarters
83  sampleGenes[1] = new IntegerGene(conf, 0, 2 * 10); // Dimes
84  sampleGenes[2] = new IntegerGene(conf, 0, 1 * 10); // Nickels
85  sampleGenes[3] = new IntegerGene(conf, 0, 4 * 10); // Pennies
86  IChromosome sampleChromosome = new Chromosome(conf, sampleGenes);
87  conf.setSampleChromosome(sampleChromosome);
88  // Finally, we need to tell the Configuration object how many
89  // Chromosomes we want in our population. The more Chromosomes,
90  // the larger number of potential solutions (which is good for
91  // finding the answer), but the longer it will take to evolve
92  // the population (which could be seen as bad).
93  // ------------------------------------------------------------
94  conf.setPopulationSize(80);
95  // Create random initial population of Chromosomes.
96  // Here we try to read in a previous run via XMLManager.readFile(..)
97  // for demonstration purpose!
98  // -----------------------------------------------------------------
99  Genotype population;
100  // Initialize the population randomly
101  population = Genotype.randomInitialGenotype(conf);
102  // Evolve the population. Since we don't know what the best answer
103  // is going to be, we just evolve the max number of times.
104  // ---------------------------------------------------------------
105  for (int i = 0; i < MAX_ALLOWED_EVOLUTIONS; i++) {
106  population.evolve();
107  }
108  // Display the best solution we found.
109  // -----------------------------------
110  IChromosome bestSolutionSoFar = population.getFittestChromosome();
111  System.out.println("The best solution has a fitness value of " +
112  bestSolutionSoFar.getFitnessValue());
113  System.out.println("It contained the following: ");
114  System.out.println("\t" +
115  DynamicMutationFitnessFunction.
116  getNumberOfCoinsAtGene(
117  bestSolutionSoFar, 0) + " quarters.");
118  System.out.println("\t" +
119  DynamicMutationFitnessFunction.
120  getNumberOfCoinsAtGene(
121  bestSolutionSoFar, 1) + " dimes.");
122  System.out.println("\t" +
123  DynamicMutationFitnessFunction.
124  getNumberOfCoinsAtGene(
125  bestSolutionSoFar, 2) + " nickels.");
126  System.out.println("\t" +
127  DynamicMutationFitnessFunction.
128  getNumberOfCoinsAtGene(
129  bestSolutionSoFar, 3) + " pennies.");
130  System.out.println("For a total of " +
131  DynamicMutationFitnessFunction.amountOfChange(
132  bestSolutionSoFar) + " cents in " +
133  DynamicMutationFitnessFunction.
134  getTotalNumberOfCoins(
135  bestSolutionSoFar) + " coins.");
136  }
