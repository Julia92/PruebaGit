809  protected static final List split(String a_string)
810  throws UnsupportedRepresentationException {
811  List a = Collections.synchronizedList(new ArrayList());
812  // No Header data.
813  // ---------------
814 
815  // Chromosome data.
816  // ----------------
817  StringTokenizer st = new StringTokenizer
818  (a_string, CHROM_DELIMITER_HEADING + CHROM_DELIMITER_CLOSING, true);
819  while (st.hasMoreTokens()) {
820  if (!st.nextToken().equals(CHROM_DELIMITER_HEADING)) {
821  throw new UnsupportedRepresentationException(a_string + " no open tag");
822  }
823  String n = st.nextToken();
824  if (n.equals(CHROM_DELIMITER_CLOSING)) {
825  a.add(""); /* Empty token */
826  }
827  else {
828  a.add(n);
829  if (!st.nextToken().equals(CHROM_DELIMITER_CLOSING)) {
830  throw new UnsupportedRepresentationException
831  (a_string + " no close tag");
832  }
833  }
834  }
835  return a;

838  /***
839  * Hashcode function for the genotype, tries to create a unique hashcode for
840  * the chromosomes within the population. The logic for the hashcode is
841  *
842  * Step Result
843  * ---- ------
844  * 1 31*0 + hashcode_0 = y(1)
845  * 2 31*y(1) + hashcode_1 = y(2)
846  * 3 31*y(2) + hashcode_2 = y(3)
847  * n 31*y(n-1) + hashcode_n-1 = y(n)
848  *
849  * @return the computed hashcode
850  *
851  * @author Klaus Meffert
852  * @since 3.5
853  */
854  public int hashCode() {
855  int i, size = size();
856  IChromosome s;
857  int twopower = 1;
858  // For empty population we want a special value different from other
859  // hashcode implementations.
860  // ------------------------------------------------------------------
861  int localHashCode = -593;
862  for (i = 0; i < size; i++, twopower = 2 * twopower) {
863  s = getChromosome(i);
864  localHashCode = 31 * localHashCode + s.hashCode();
865  }
866  return localHashCode;
867  }
